1Corinthians

Chapter 1

פּוֹלוֹס שָׁלִיחַ מְקׂרָא שֶׁל־יֵשׁוּעַ הַמָּשִׁיחַ בִּרְצוֹן אֱלׂהִים וְסוֹסְתְּנִיס אָחִינוּ׃

א

1

אֶל־קְהִלַּת אֱלׂהִים אֲשֶׁר בְּקוֹרִנְתּוֹס לַמְקֻדָּשִׁים בַּמָּשִׁיחַ יֵשׁוּעַ הַקְּדשִׁים הַקְּרוּאִים עִם כָּל־הַקּׂרְאִים בְּשֵׁם אֲדׂנֵינוּ יֵשׁוּעַ הַמָּשִׁיחַ בְּכָל־מָקוֹם שֶׁלָּהֶם וְשֶׁלָּנוּ׃

ב

2

חֶסֶד לָכֶם וְשָׁלוֹם מֵאֵת הָאֱלׂהִים אָבִינוּ וַאֲדׂנֵינוּ יֵשׁוּעַ הַמָּשִׁיחַ׃

ג

3

אוֹדֶה לֵאלׂהַי בַּעֲבוּרְכֶם בְּכָל־עֵת עַל־חֶסֶד הָאֱלׂהִים הַנִּתָּן לָכֶם בַּמָּשִׁיחַ יֵשׁוּעַ׃

ד

4

אֲשֶׁר עֲשַׁרְתֶּם־בּוֹ בַּכּׂל בְּדִבּוּר וּבְדָעַת׃

ה

5

בַּאֲשֶׁר עֵדוּת הַמָּשִׁיחַ הִתְחַזְּקָה בָכֶם׃

ו

6

עַד אֲשֶׁר־לׂא חֲסַרְתֶּם כָּל־מַתַּן הֶחָסֶד וְהִנְּכֶם מְחַכִּים לְהִתְגַּלּוּת אֲדׂנֵינוּ יֵשׁוּעַ הַמָּשִׁיחַ׃

ז

7

וְהוּא גַּם־יְכוֹנֵן אֶתְכֶם עַד־עֵת קֵץ לִהְיוֹת נְקִיִּים בְּיוֹם אֲדׂנֵינוּ יֵשׁוּעַ הַמָּשִׁיחַ׃

ח

8

נֶאֱמָן הָאֱלׂהִים אֲשֶׁר עַל־פִּיו נִקְרֵאתֶם לְחֶבְרַת בְּנוֹ יֵשׁוּעַ הַמָּשִׁיחַ אֲדׂנֵינוּ׃

ט

9

וְהִנְנִי מַזְהִיר אֶתְכֶם אַחַי בְּשֵׁם אֲדׂנֵינוּ יֵשׁוּעַ הַמָּשִׁיחַ לִהְיוֹת כֻּלְּכֶם פֶּה אֶחָד וְלׂא תִּהְיֶינָה מַחֲלׂקוֹת בֵּינֵיכֶם כִּי אִם־תִּכּוֹנֲנוּ יַחַד בְּלֵב אֶחָד וּבְעֵצָה אֶחָת׃

י

10

כִּי בְּנֵי־בֵית כְּלוֹאָה הִגִּידוּ לִי עֲלֵיכֶם אֶחָי כִּי מְרִיבוֹת בֵּינֵיכֶם׃

יא

11

וְזׂאת אֲנִי אׂמֵר מִפְּנֵי שֶׁאִישׁ אִישׁ מִכֶּם אׂמֵר אֲנִי לְפוֹלוֹס וַאֲנִי לְאַפּוֹלוֹס וַאֲנִי לְכֵיפָא וַאֲנִי לַמָּשִׁיחַ הִנֵּנִי׃

יב

12

הֲכִי חֻלַּק הַמָּשִׁיחַ הֲכִי פוֹלוֹס נִצְלַב בַּעַדְכֶם אִם לְשֵׁם פּוֹלוֹס נִטְבַּלְתֶּם׃

יג

13

אוֹדֶה לֵאלׂהִים שֶׁלּא טָבַלְתִּי אִישׁ מִכֶּם בִּלְתִּי קְרִסְפּוֹס וְגָיוֹס׃

יד

14

פֶּן־יׂאמְרוּ כִּי־לִשְׁמִי טָבָלְתִּי׃

טו

15

וְטָבַלְתִּי גַּם אֶת־בְּנֵי־בֵית אִסְטְפָנוֹס וּמִלְּבַד אֵלֶּה לׂא אֵדַע אִם־טָבַלְתִּי אִישׁ אַחֵר׃

טז

16

כִּי לׂא שְׁלָחַנִי הַמָּשִׁיחַ לִטְבּוֹל כִּי אִם־לְבַשֵּׂר לׂא־בְחָכְמַת דְּבָרִים לְמַעַן אֲשֶׁר לׂא־יִהְיֶה צְלוּב הַמָּשִׁיחַ לָרִיק׃

יז

17

כִּי־דְבַר הַצְּלוּב סִכְלוּת הוּא לָאׂבְדִים אֲבָל לָנוּ הַנּוֹשָׁעִים גְּבוּרַת אֱלׂהִים׃

יח

18

כִּי־כֵן כָּתוּב אֲאַבֵּד חָכְמַת חֲכָמִים וּבִינַת נְבוֹנִים אַסְתִּיר׃

יט

19

אַיֵּה חָכָם אַיֵּה סׂפֵר אַיֵּה דּׂרֵשׁ הָעוֹלָם הַזֶּה הֲלׂא סִכֵּל הָאֳלׂהִים אֶת־חָכְמַת הָעוֹלָם הַזֶּה׃

כ

20

כִּי אַחֲרֵי אֲשֶׁר בְּחָכְמַת הָאֱלׂהִים לׂא יָדַע הָעוֹלָם אֶת־הָאֱלׂהִים בְּחָכְמָה הָיָה רָצוֹן מִלְּפָנָיו לְהוֹשִׁיעַ בְּסִכְלוּת הַקְּרִיאָה אֶת הַמַּאֲמִינִים׃

כא

21

כִּי הַיְּהוּדִים שׁׂאֲלִים לָהֶם אוֹת וְהַיְּוָנִים מְבַקְשִׁים חָכְמָה׃

כב

22

וַאֲנַחְנוּ מַשְׁמִיעִים אֶת־הַמָּשִׁיחַ הַצָּלוּב מִכְשׁוֹל לַיְהוּדִים וְסִכְלוּת לַיְּוָנִים׃

כג

23

אֲבָל לַמְקׂרָאִים בֵּין מִיהוּדִים בֵּין מִיוָנִים אֶת־הַמָּשִׁיחַ אֲשֶׁר הוּא גְּבוּרַת אֱלׂהִים וְחָכְמַת אֱלׂהִים׃

כד

24

יַעַן כִּי סִכְלוּת הָאֵל חֲכָמָה הִיא מֵאָדָם וְחֻלְשַׁת הָאֵל חֲזָקָה הִיא מֵאָדָם׃

כה

25

כִּי רְאוּ־נָא אַחַי אֶת־קְרִיאַתְכֶם שֶׁלּא רַבִּים הַחֲכָמִים מִן־הַבָּשָׂר לׂא רַבִּים הַשַּׁלִּיטִים לׂא רַבִּים רָמֵי הַיַּחַשׂ׃

כו

26

כִּי אִם־בַּסָּכָל שֶׁבָּעוֹלָם בָּחַר הָאֵל לְמַעַן בַּיֵּשׁ אֶת־הַחֲכָמִים וּבֶחָלוּשׁ שֶׁבָּעוֹלָם בָּחַר הָאֵל לְמַעַן בַּיֵּשׁ אֶת־הֶחָזָק׃

כז

27

וּבְדַלַּת הָעוֹלָם בָּחַר הָאֵל וּבַנִּמְאָס וּבַאֲשֶׁר כְּאָיִן לְמַעַן בַּטֵּל אֵת אֲשֶׁר יֶשְׁנוֹ׃

כח

28

כְּדֵי שֶׁלּא־יִתְהַלֵּל לְפָנָיו כָּל־בָּשָׂר׃

כט

29

וּמִמֶּנּוּ אַתֶּם בַּמָּשִׁיחַ יֵשׁוּעַ אֲשֶׁר הָיָה לָנוּ לְחָכְמָה מֵאֵת הָאֱלׂהִים וְלִצְדָקָה וְלִקְדֻשָּׁה וּלְפִדְיוֹם׃

ל

30

וִיהִי כַּכָּתוּב הַמִּתְהַלֵּל יִּתְהַלֵּל בַּיְהוָֹה׃

לא

Chapter 2

וְגַם אָנׂכִי בְּבֹאִי אֲלֵיכֶם אַחַי לׂא בָאתִי בְּגֵאוּת הַדִבּוּר וְהַחָכְמָה לְהַגִּיד לָכֶם אֶת־עֵדוּת הָאֱלׂהִים׃

א

1

כִּי לׂא־אָמַרְתִּי לָדַעַת בְּתוֹכֲכֶם דָּבָר בִּלְתִּי אִם־יֵשׁוּעַ הַמָּשִׁיחַ וְהוּא הַנִּצְלָב׃

ב

2

וָאֶהִי עִמָּכֶם בְּחֻלְשָׁה וּבְיִרְאָה וּבְחַלְחָלָה רַבָּה׃

ג

3

וּדְבָרִי וּקְרִיאָתִי לׂא לְפַתּוֹת בְּאִמְרֵי חָכְמַת בְּנֵי־אָדָם כִּי אִם־בְּתוֹכַחַת הָרוּחַ וְהַגְּבוּרָה׃

ד

4

לְמַעַן אֲשֶׁר לׂא־תִהְיֶה אֱמוּנַתְכֶם בְּחָכְמַת בְּנֵי אָדָם כִּי אִם־בִּגְבוּרַת אֱלׂהִים׃

ה

5

אֲבָל חָכְמָה אֲנַחְנוּ מְדַבְּרִים בְּקֶרֶב הַשְּׁלֵמִים לׂא חָכְמַת הָעוֹלָם הַזֶּה גַּם־לׂא שֶׁל־שָׂרֵי הָעוֹלָם הַזֶּה אֲשֶׁר יׂאבֵדוּ׃

ו

6

כִּי אִם־נְדַבֵּר בַּסּוֹד חָכְמַת הָאֱלׂהִים הַנִּסְתָּרָה אֲשֶׁר הָאֱלׂהִים יְעָדָהּ לִכְבוֹדֵנוּ לִפְנֵי יְמוֹת עוֹלָם׃

ז

7

אֲשֶׁר לׂא יְדָעָהּ אִישׁ מִשָּׂרֵי הָעוֹלָם הַזֶּה כִּי אִלּוּ יְדָעוּהָ לׂא צָלְבוּ אֶת־אֲדוֹן הַכָּבוֹד׃

ח

8

כִּי אִם־כַּכָּתוּב אֲשֶׁר־עַיִן לׂא רָאֲתָה וְאׂזֶן לׂא שָׁמְעָה וְלׂא עָלָה עַל־לֵב אָדָם אֵת אֲשֶׁר־הֵכִין הָאֱלׂהִים לְאׂהֲבָיו׃

ט

9

וְלָנוּ גִּלָּה הָאֱלׂהִים בְּרוּחוֹ כִּי הָרוּחַ חוֹקֵר אֶת־הַכּׂל גַּם אֶת־מַעֲמַקֵּי הָאֱלׂהִים׃

י

10

כִּי מִי הוּא מִבְּנֵי אָדָם יׂדֵעַ אֵת אֲשֶׁר בָּאָדָם כִּי אִם־רוּחַ הָאָדָם אֲשֶׁר בְּקִרְבּוֹ וְכֵן אֵין אִישׁ יׂדֵעַ אֵת אֲשֶׁר בֵּאלׂהִים כִּי אִם־רוּחַ הָאֱלׂהִים׃

יא

11

וַאֲנַחְנוּ לׂא לָקַחְנוּ אֶת־רוּחַ הָעוֹלָם כִּי אִם־הָרוּחַ מֵאֵת הָאֱלׂהִים לְמַעַן נֵדַע אֶת־אֲשֶׁר נִתַּן לָנוּ מֵאֵת הָאֱלׂהִים בְּחַסְדּוֹ׃

יב

12

וְאֶת־זׂאת לׂא נוֹצִיא בְּלָשׁוֹן אֲשֶׁר תְּלַמֵּד חָכְמַת בְּנֵי אָדָם כִּי אִם־בְּלָשׁוֹן אֲשֶׁר רוּחַ הַקּׂדֶשׁ תְּלַמֵּד וּנְבָאֵר רוּחָנִיּוֹת בִּדְבָרִים רוּחַנִיּים׃

יג

13

הֵן הָאָדָם הַטִּבְעִי אֵינֶנּוּ מְקַבֵּל אֶת־דִּבְרֵי רוּחַ אֱלׂהִים כִּי־הֵמָּה סִכְלוּת לוֹ וְלׂא יוּכַל לַהֲבִינָם בַּאֲשֶׁר הֵמָּה נְדוֹנִים בְּדֶרֶךְ הָרוּחַ׃

יד

14

אֲבָל הָאָדָם הָרוּחָנִי יָדִין הַכּׂל וְאוֹתוֹ לׂא־יָדִין אִישׁ׃

טו

15

כִּי מִי־תִכֵּן אֶת־רוּחַ יְהוָֹה וּמִי יוֹדִיעֶנּוּ וַאֲנַחְנוּ הִנֵּה יֶשׁ־לָנוּ רוּחַ הַמָּשִׁיחַ׃

טז

16

Chapter 3

וַאֲנִי לׂא יָכׂלְתִּי לְדַבֵּר עִמָּכֶם אַחַי כְּדַבֵּר עִם אַנְשֵׁי רוּחַ כִּי אִם כְּדַבֵּר עִם אַנְשֵׁי בָשָׂר כְּדַבֵּר עִם עוֹלֲלִים בַּמָּשִׁיחַ׃

א

1

חָלָב הִשְׁקֵיתִי אֶתְכֶם וְלׂא־מַאֲכָל כִּי עֲדַיִן לׂא הֱיִיתֶם יְכוֹלִים וְגַם־עַתָּה אֵינְכֶם יְכוֹלִים כִּי עוֹדְכֶם שֶׁל־הַבָּשָׂר׃

ב

2

כִּי בַּאֲשֶׁר קִנְאָה בֵּינֵיכֶם וּמְרִיבָה וּמַחֲלׂקוֹת הֲלׂא שֶׁל־הַבָּשָׂר אַתֶּם וְנׂהֲגִים מִנְהַג בְּנֵי אָדָם׃

ג

3

הֵן בֶּאֱמׂר זֶה אֲנִי לְפוֹלוֹס וְזֶה אֲנִי לְאַפּוֹלוֹס הֲלׂא שֶׁל־הַבָּשָׂר אַתֶּם׃

ד

4

מִי אֵפוֹא פוֹלוֹס וּמִי־הוּא אַפּוֹלוֹס אַךְ־מְשָׁרֲתִים אֲשֶׁר עַל־יָדָם בָּאתֶם לְהַאֲמִין אִישׁ אִישׁ כְּמַתְּנַת הָאָדוֹן אֲשֶׁר נָתַן לוֹ׃
