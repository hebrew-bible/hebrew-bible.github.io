1Thessalonians

Chapter 1

פּוֹלוֹס וְסִלְוָנוֹס וְטִימוֹתִיּוֹס אֶל־קְהִלַּת הַתֶּסָּלוֹנִיקִיּים בֵּאלׂהִים הָאָב וּבָאָדוֹן יֵשׁוּעַ הַמָּשִׁיחַ חֶסֶד לָכֶם וְשָׁלוֹם מֵאֵת אֱלׂהִים אָבִינוּ וַאֲדׂנֵינוּ יֵשׁוּעַ הַמָּשִׁיחַ׃

א

1

נוֹדֶה לֵאלׂהִים עַל־כֻּלְּכֶם בְּכָל־עֵת בְּהַזְכִּירֵנוּ אֶתְכֶם בִּתְפִלּוֹתֵינוּ׃

ב

2

בְּזָכְרֵנוּ תָמִיד לִפְנֵי הָאֱלׂהִים אָבִינוּ אֶת־פּׂעַל אֱמוּנַתְכֶם וִיגִיעַת אַהֲבַתְכֶם וְסַבְלָנוּת תִּקְוַתְכֶם לַאֲדׂנֵינוּ יֵשׁוּעַ הַמָּשִׁיחַ׃

ג

3

כִּי יָדַעְנוּ אַחַי חֲבִיבֵי הָאֱלׂהִים אֵת אֲשֶׁר נִבְחַרְתֶּם׃

ד

4

בַּאֲשֶׁר בְּשׂוֹרָתֵנוּ לׂא בָאָה לָכֶם בְּדִבּוּר לְבָד כִּי־גַם בִּגְבוּרָה וּבְרוּחַ הַקּׂדֶשׁ וּבְדַעַת נְכוֹנָה מְאׂד כַּאֲשֶׁר יְדַעְתֶּם אֵת אֲשֶׁר־הָיִינוּ בְּתוֹכֲכֶם לְמַעַנְכֶם׃

ה

5

וִהְיִיתֶם לְהׂלְכֵי בְעִקְּבוֹתֵינוּ וּבְעִקְּבוֹת אֲדׂנֵינוּ כִּי קִבַּלְתֶּם אֶת־הַדָּבָר בְּתוֹךְ עֳנִי רָב בְּחֶדְוַת רוּחַ הַקּׂדֶשׁ׃

ו

6

עַד־הֱיוֹתְכֶם לְמוֹפֵת לְכָל־הַמַּאֲמִינִים אֲשֶׁר בְּמַקְדּוֹנְיָא וּבַאֲכַיָּא׃

ז

7

כִּי מֵאִתְּכֶם נִשְׁמַע קוֹל דְּבַר יְהוָֹה לׂא בְּמַקְדּוֹנְיָא וַאֲכַיָּא לְבַד כִּי בְּכָל־מָקוֹם יָצְאָה אֱמוּנַתְכֶם לֵאלׂהִים וְאֵין־לָנוּ צׂרֶךְ לְדַבֵּר עוֹד דָּבָר׃

ח

8

כִּי־פִיהֶם הַמַּגִּיד מֶה־הָיָה מְבוֹאֵנוּ אֲלֵיכֶם וְאֵיךְ עֲזַבְתֶּם הָאֱלִילִים וְשַׁבְתֶּם לֵאלׂהִים לַעֲבֹד אֵל חַי וַאֲמִתִּי׃

ט

Chapter 2

כִּי אַתֶּם אַחַי יְדַעְתֶּם אֶת־מְבוֹאֵנוּ אֲלֵיכֶם כִּי לׂא־הָיָה לָרִיק׃

א

1

וְאַחֲרֵי אֲשֶׁר עֻנֵּינוּ וּלְחֶרְפָּה הָיִינוּ בְּפִילִפִּי כַּאֲשֶׁר יְדַעְתֶּם הִתְחַזַּקְנוּ בֵאלׂהֵינוּ לְהַגִּיד גַּם־לָכֶם אֶת־הַבְּשׂוֹרָה בְּנַפְתּוּלִים רַבִּים׃

ב

2

כִּי תוֹכַחְתֵּנוּ אֵינֶנָּה מִתּוֹךְ טָעוּת וְלׂא מִתּוֹךְ טֻמְאָה וְלׂא בִּרְמִיָּה׃

ג

3

כִּי כַּאֲשֶׁר נִמְצֵאנוּ נֶאֱמָנִים לֵאלׂהִים לְהַפְקִיד בְּיָדֵנוּ הַבְּשׂוֹרָה כֵּן נְדַבֵּר וְלׂא לִהְיוֹת רְצוּיִם לִבְנֵי אָדָם כִּי אִם־לֵאלׂהִים הַבֹּחֵן לִבּוֹתֵנוּ׃

ד

4

כִּי מֵעוֹלָם לׂא דִבַּרְנוּ בִּשְׂפַת חֲלָקוֹת כַּאֲשֶׁר יְדַעְתֶּם וְלׂא לְמַעַן בְּצׂעַ בֶּצַע בַּסָּתֶר עֵד הָאֱלׂהִים׃

ה

5

גַּם לׂא־בִקַּשְׁנוּ מִן־הָאָדָם כָּבוֹד לׂא מִכֶּם וְלׂא מֵאֲחֵרִים אַף כִּי יָכׂלְנוּ לִהְיוֹת בִּיקָר כִּשְׁלִיחֵי הַמָּשִׁיחַ׃

ו

6

אֶלָּא הָלַכְנוּ לְאַט בְּתוֹכֲכֶם כְּאׂמֶנֶת מְפַנֶּקֶת אֶת־בָּנֶיהָ׃

ז

7

וּבְחַבְּבֵנוּ אֶתְכֶם כָּכָה חָפַצְנוּ לָתֵת לָכֶם לׂא אֶת־בְּשׂוֹרַת הָאֱלׂהִים לְבַד אֶלָּא אַף נַפְשׁׂתֵינוּ יַעַן כִּי־הֱיִיתֶם חֲבִיבִים עָלֵינוּ׃

ח

8

הֲלׂא תִזְכְּרוּ אַחַי אֶת־עֲמָלֵנוּ וְאֶת־תְּלָאָתֵנוּ אֲשֶׁר יָגַעְנוּ לַיְלָה וְיוֹמָם לְבִלְתִּי הֱיוֹת לְמַשָׂא לְאִישׁ מִכֶּם בְּבַשְׂרֵנוּ בְקִרְבְּכֶם אֶת־בְּשׂוֹרַת הָאֱלׂהִים׃

ט

9

עֵדִים אַתֶּם וְעֵד הָאֱלׂהִים אִם־לׂא בִקְדֻשָּׁה וּבְצֶדֶק וּבְתָמִים הָיִינוּ עִמָּכֶם הַמַּאֲמִינִים׃

י

10

וְאַתֶּם יְדַעְתֶּם כִּי־כְאָב אֶת־בָּנָיו הִזְהַרְנוּ אֶת־כָּל־אֶחָד מִכֶּם וְדִבַּרְנוּ עַל לִבּוֹ׃

יא

11

וַנָּעַד בָּכֶם לָלֶכֶת כַּטּוֹב לִפְנֵי הָאֱלׂהִים הַקּוֹרֵא אֶתְכֶם לְמַלְכוּתוֹ וְלִכְבוֹדוֹ׃

יב

12

בַּעֲבוּר זׂאת אֲנַחְנוּ לׂא נֶחְדַּל לְהוֹדׂת לֵאלׂהִים כִּי אַתֶּם קִבַּלְתֶּם מֵאִתָּנוּ דְּבַר שְׁמוּעַת הָאֱלׂהִים הַמַּאֲמִינִים׃

יג

13

כִּי־אַתֶּם אַחַי הֲלַכְתֶּם בְּעִקְּבֵי קְהִלּוֹת הָאֱלׂהִים אֲשֶׁר־בְּאֶרֶץ יְהוּדָה בַּמָּשִׁיחַ יֵשׁוּעַ כִּי גַּם־אַתֶּם עֻנֵּיתֶם עַל־יְדֵי בְנֵי־עַמְּכֶם כַּאֲשֶׁר עֻנּוּ גַם־הֵם עַל־יְדֵי הַיְּהוּדִים׃

יד

14

אֲשֶׁר אַף־הָרְגוּ אֶת־הָאָדוֹן יֵשׁוּעַ וְאֶת־נְבִיאֵיהֶם וְאוֹתָנוּ רָדָפוּ וְאֵינָם טוֹבִים בְּעֵינֵי אֱלׂהִים וּמִתְקוֹמֲמִים לְכָל־אָדָם׃

טו

15

הָעׂצְרִים אׂתָנוּ מִדַּבֵּר אֶל־הַגּוֹיִם שֶׁיִּוָּשֵׁעוּ לְמַעַן אֲשֶׁר־יְמַלְאוּ אֶת־חַטּׂאתֵיהֶם בְּכָל־עֵת וַתַּשִׂיגֵם הַחֵמָה עַד־כַּלֵּה׃

טז

16

וַאֲנַחְנוּ אַחַי אַחֲרֵי אֲשֶׁר־שָׁכׂלְנוּ אֶתְכֶם לִזְמַן מְעַט לְפָנִים וְלׂא לַלֵּב חַשְׁנוּ עַד־מְאׂד לִרְאוֹת פְּנֵיכֶם בִּתְשׁוּקָה רַבָּה׃

יז

17

וְעַל־כֵּן חָפַצְנוּ לָבוֹא אֲלֵיכֶם אֲנִי פוֹלוֹס פַּעַם וּשְׁתָּיִם וְהַשָׂטָן עֲצָרָנוּ׃

יח

18

כִּי מִי תִקְוָתֵנוּ וּמִי שִׂמְחָתֵנוּ וַעֲטֶרֶת תִּפְאַרְתֵּנוּ הֲלׂא גַם־אַתֶּם לִפְנֵי אֲדׂנֵינוּ יֵשׁוּעַ הַמָּשִׁיחַ בְּבוֹאוֹ׃

יט

19

אָמְנָם אַתֶּם כְּבוֹדֵנוּ וְשִׂמְחָתֵנוּ׃

Chapter 3

וְעַל־כֵּן לׂא יָכׂלְנוּ עוֹד לְהִתְאַפֵּק וַנִּוָּעַץ לְהִוָּתֵר לְבַדֵּנוּ בְּאַתּוּנָה׃

א

1

וַנִּשְׁלַח אֶת־טִימוֹתִיּוֹס אָחִינוּ וּמְשָׁרֵת הָאֱלׂהִים וְעׂזְרֵנוּ בִּבְשׂוֹרַת הַמָּשִׁיחַ לְחַזֵּק אֶתְכֶם וּלְהַזְהִיר עַל־דְּבַר אֱמוּנַתְכֶם׃

ב

2

לְמַעַן אֲשֶׁר לׂא־יִמּוֹט אִישׁ בַּמְּצוּקוֹת הָאֵלֶּה כִּי־יְדַעְתֶּם אַף־אַתֶּם כִּי לָזׂאת יֻעָדְנוּ׃

ג

3

הֲלׂא כְּבָר אָמַרְנוּ אֲלֵיכֶם בִּהְיוֹתֵנוּ אֶצְלְכֶם כִּי־עֲתִידִים אֲנַחְנוּ לְהִלָּחֵץ כַּאֲשֶׁר גַּם־הָיָה וְכַאֲשֶׁר יְדַעְתֶּם׃

ד

4

וּבַעֲבוּר כֵּן לׂא־יָכׂלְתִּי עוֹד לְהִתְאַפֵּק וָאֶשְׁלַח לָדַעַת אֱמוּנַתְכֶם כִּי אָמַרְתִּי פֶּן־נִסָּה אֶתְכֶם הַמְנַסֶּה וְהָיְתָה לָרִיק יְגִיעָתֵנוּ׃

ה

5

וְעַתָּה כְּשׁוּב אֵלֵינוּ טִימוֹתִיּוֹס מֵאִתְּכֶם וַיְבַשֵׂר אׂתָנוּ אֶת־בְּשׂוֹרַת אֱמוּנַתְכֶם וְאַהֲבַתְכֶם וְאֵת אֲשֶׁר אַתֶּם זׂכְרִים אׂתָנוּ לְטוֹבָה בְּכָל־עֵת וְנִכְסָפִים לִרְאוֹת אׂתָנוּ כַּאֲשֶׁר נִכְסׂף גַּם־אֲנַחְנוּ לִרְאוֹתְכֶם׃

ו

6

אָז נֻחַמְנוּ עֲלֵיכֶם אַחַי בְּכָל־צָרָתֵנוּ וְלַחֲצֵנוּ עַל־יְדֵי אֱמוּנַתְכֶם׃

ז

7

כִּי עַתָּה נִחְיֶה אִם־תַּעַמְדוּ אַתֶּם בַּאֲדׂנֵינוּ׃

ח

8

כִּי מַה־נָּשִׁיב לֵאלׂהִים לְהוֹדוֹת לוֹ בַּעֲבוּרְכֶם עַל כָּל־הַשִׂמְחָה אֲשֶׁר שָׂמַחְנוּ עֲלֵיכֶם לִפְנֵי אֱלׂהֵינוּ׃

ט

9

לַיְלָה וְיוֹמָם מִתְפַּלְּלִים אֲנַחְנוּ בְּרָב־תַּחֲנוּנִים לִרְאוֹת אֶת־פְּנֵיכֶם וּלְהַשְׁלִים אֶת־מַחֲסׂרֵי אֱמוּנַתְכֶם׃

י

10

וְהָאֱלׂהִים אָבִינוּ וַאֲדׂנֵינוּ יֵשׁוּעַ הַמָּשִׁיחַ הוּא יְיַשֵּׁר אֶת־דַּרְכֵּנוּ אֲלֵיכֶם׃

יא

11

וְאֶתְכֶם יַרְבֶּה הָאָדוֹן וְיַפְרֶה בְּאַהֲבַת אִישׁ אֶת־רֵעֵהוּ וּבְאַהֲבַת כָּל־אָדָם כַּאֲשֶׁר אֲנַחְנוּ אׂהֲבִים אֶתְכֶם׃

יב

12

לְחַזֵּק אֶת־לְבַבְכֶם לִהְיוֹת תָּמִים בִּקְדֻשָּׁה לִפְנֵי אֱלׂהֵינוּ אָבִינוּ בְּבוֹא יֵשׁוּעַ הַמָּשִׁיחַ אֲדׂנֵינוּ עִם־כָּל־קְדשָׁיו׃

Chapter 4

וְעוֹד נְבַקְשָׁה מִכֶּם אַחַי וְנַזְהִיר אֶתְכֶם בַּאֲדׂנֵינוּ יֵשׁוּעַ כִּי כַּאֲשֶׁר קִבַּלְתֶּם מֵאִתָּנוּ אֵיךְ עֲלֵיכֶם לְהִתְהַלֵּךְ וְלִמְצׂא חֵן בְּעֵינֵי הָאֱלׂהִים וְכַאֲשֶׁר גַּם־הִתְהַלַּכְתֶּם כֵּן תּוֹסִיפוּ וְכֵן תַּרְבּוּ עוֹד׃

כִּי זֶה רְצוֹן הָאֱלׂהִים קְדֻשַּׁתְכֶם לְהִתְרַחֵק מִן־הַזְּנוּת׃

3

וְיֵדַּע כָּל־אִישׁ מִכֶּם לְהִשְׁתַּמֵּשׁ בְּכִלְיוֹ בִּקְדֻשָּׁה וִיקָר׃

וְלׂא־בְתַאֲוַת זִמָּה כְּדֶרֶךְ הַגּוֹיִם אֲשֶׁר־אֵינָם יׂדְעִים הָאֱלׂהִים׃

8

ט

9

וְגַם עֲשִׂיתֶם כֵּן לְכָל־אֲחֵיכֶם אֲשֶׁר בְּכָל־מַקְדּוֹנְיָא אֲבָל נְבַקְשָׁה מִכֶּם אַחַי אֲשֶׁר כֵּן תַּעֲשׂוּ וְכֵן תּוֹסִיפוּ׃

יא

11

לְמַעַן תִּתְהַלְּכוּ כְהׂגֶן עִם־אׂתָם שֶׁבַּחוּץ וְלׂא תִּצְטָרֲכוּ לְאָדָם׃

יב

וְעַל־דְּבַר הַיְשֵׁנִים אֶחָי לׂא־נְכַחֵד מִכֶּם דָּבָר לְמַעַן לׂא תֵּעָצְבוּ כַּאֲחֵרִים אֲשֶׁר אֵין־לָהֶם תִּקְוָה׃

יג

13

כִּי אִם־נַאֲמִין אֲשֶׁר־מֵת יֵשׁוּעַ וַיֶּחִי כֵּן יָבִיא הָאֱלׂהִים גַּם אֶת־הַיְשֵׁנִים בְּיֵשׁוּעַ אִתּוֹ׃

יד

14

כִּי אֶת־זׂאת נׂאמַר לָכֶם בִּדְבַר יְהוָֹה כִּי אֲנַחְנוּ הַחַיִּים הַנּוֹתָרִים עַד־בֹּא הָאָדוֹן לׂאׂ נְקַדֵּם אֶת־הַיְשֵׁנִים׃

טו

15

כִּי הָאָדוֹן הוּא יֵרֵד מִן־הַשָּׁמַיִם בִּתְרוּעָה בְּקוֹל שַׂר הַמַּלְאָכִים וּבְשׁוֹפַר אֱלׂהִים וְאָז יָקוּמוּ רִאשׁוֹנָה הַמֵּתִים בַּמָּשִׁיחַ׃

טז

16

אַחֲרֵי־כֵן אֲנַחְנוּ הַחַיִּים הַנּוֹתָרִים נִלָּקַח אִתָּם יַחְדָּו בַּעֲנָנִים לִקְרַאת הָאָדוֹן לָאֲוִיר וּבְכֵן נִהְיֶה תָמִיד עִם־הָאָדוֹן׃

יז

17

לָכֵן נַחֲמוּ זֶה אֶת־זֶה בַּדְּבָרִים הָאֵלֶּה׃

Chapter 5

וְעַל־דְּבַר הָעִתִּים וְהַזְּמַנִּים אֵין־צׂרֶךְ לִכְתּׂב אֲלֵיכֶם אֶחָי׃

א

1

הֲלׂא אַף־אַתֶּם יְדַעְתֶּם הֵיטֵב כִּי יוֹם־יְהוָֹה כַּגַּנָּב בַּלַּיְלָה כֵּן יָבוֹא׃

ב

2

כִּי בְאָמְרָם שָׁלוֹם וְשַׁלְוָה יָבֹא עֲלֵיהֶם הַשֶּׁבֶר פִּתְאׂם כְּחֶבְלֵי יוֹלֵדָה וְלׂא יוּכְלוּ הִמָּלֵט׃

ג

3

אֲבָל אַתֶּם אַחַי לׂא בַּחשֶׁךְ אַתֶּם שֶׁיַּשִׂיג אֶתְכֶם הַיּוֹם כַּגַּנָּב׃

ד

4

אַתֶּם כֻּלְּכֶם בְּנֵי הָאוֹר וּבְנֵי הַיּוֹם לׂא בְנֵי־הַלַּיְלָה אֲנַחְנוּ וְלׂא בְּנֵי הַחשֶׁךְ׃

ה

5

לָכֵן אַל־נָא נֵרָדֵם כְּמוֹ הָאֲחֵרִים כִּי אִם־נִשְׁקׂד וְנִהְיֶה ְמְפֻכָּחִים׃

ו

6

כִּי הַנִּרְדָּמִים נִרְדָּמִים בַּלָּיְלָה וְהַמִּשְׁתַּכְּרִים מִשְׁתַּכְּרִים בַּלָּיְלָה׃

ז

וַאֲנַחְנוּ בְּנֵי הַיּוֹם נִתְּפַּכֵּחַ־נָּא וְנִלְבַּשׁ אֶת־שִׁרְיוֹן הָאֱמוּנָה וְהָאַהֲבָה וְכַכּוֹבַע אֶת־תִּקְוַת הַיְשׁוּעָה׃

ח

יַעַן אֲשֶׁר לׂא־יְעָדָנוּ הָאֱלׂהִים לְחָרוֹן כִּי לָרֶשֶׁת אֶת־הַיְשׁוּעָה עַל־יְדֵי אֲדׂנֵינוּ יֵשׁוּעַ הַמָּשִׁיחַ׃
