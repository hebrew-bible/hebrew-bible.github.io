Chapter 1

פּוֹלוֹס שְׁלִיחַ יֵשׁוּעַ הַמָּשִׁיחַ עַל־פִּי מִצְוַת הָאֱלׂהִים מוֹשִׁיעֵנוּ וְהָאָדוֹן יֵשׁוּעַ הַמָּשִׁיחַ תִּקְוָתֵנוּ׃

א

1

אֶל־טִימוֹתִיּוֹס בְּנוֹ הָאֲמִתִּי בָּאֱמוּנָה חֶסֶד וְרַחֲמִים וְשָׁלוֹם מֵאֵת אֱלׂהִים אָבִינוּ וְהַמָּשִׁיחַ יֵשׁוּעַ אֲדׂנֵינוּ׃

ב

2

הִנֵּה בְּלֶכְתִּי לְמַקְדּוֹנְיָא בִּקַּשְׁתִּי מִמְּךָ לָשֶׁבֶת בְּאֶפְסוֹס לְמַעַן תְּצַוֶּה מִקְצָת אֲנָשִׁים לְבִלְתִּי הוֹרׂת תּוֹרָה זָרָה׃

ג

3

וְלׂא יָשִׂימוּ לִבָּם לְהַגָּדוֹת וּלְלִמּוּדֵי תוֹלָדוֹת שֶׁאֵין־לָהֶם סוֹף הַמְבִיאִים לִשְׁאֵלוֹת וְלׂא לִבְנוֹת בֵּית־אֱלׂהִים בָּאֱמוּנָה׃

ד

4

כִּי־תַכְלִית הַמִּצְוָה הִיא הָאַהֲבָה בְּלֵב טָהוֹר וּבְדַעַת נְקִיָּה וּבֶאֱמוּנָה לׂא־צְבוּעָה׃

ה

5

יֵשׁ אֲשֶׁר תָּעוּ מִּדַּרְכָּהּ וַיִּפְנוּ אַחֲרֵי לִמּוּדֵי הָבֶל׃

ו

6

חֲפֵצִים לִהְיוֹת מוֹרֵי תוֹרָה וְאֵינָם יׂדְעִים מָה הֵם אׂמְרִים וּמָה הֵם דָּנִים עָלָיו׃

ז

7

אֲבָל יָדַעְנוּ כִּי הַתּוֹרָה טוֹבָה אִם־יִשְׁתַּמֵּשׁ בָּהּ הָאָדָם כַּתּוֹרָה׃

ח

8

בְּדַעְתּוֹ זׂאת שֶׁהַחׂק לׂא הוּשַׂם לַצַּדִּיק כִּי אִם בַּעֲבוּר הַפּשְׁעִים וְהַמּׂרְדִים הָרְשָׁעִים וְהַחַטָּאִים טְמֵאִים וּנְבָלִים מַכֵּי־אָב וּמַכֵּי־אֵם וּמְרַצְּחִים׃

ט

9

וְזׂנִים וְשׁׂכְבִים אֶת־זָכָר וְגׂנְבֵי נֶפֶשׁ וְשַׁקְרָנִים וְנִשְׁבָּעִים לַשָּׁקֶר וְכָל־מַעֲשֶׂה הֶפֶךְ מִן־הַלֶּקַח הַבָּרִיא׃

י

10

כְּפִי בְשׂוֹרַת כְּבוֹד הַאֵל הַמְבֹרָךְ אֲשֶׁר הִפְקִיד בְּיָדִי׃

יא

11

וְהִנְנִי מוֹדֶה לְיֵשׁוּעַ הַמָּשִׁיחַ אֲדׂנֵינוּ הַמְאַזְּרֵנִי חָיִל אֲשֶׁר רָאַנִי נֶאֱמָן וַיְשִׂימֵנִי לִמְשָׁרֵת לוֹ׃

יב

12

אֲשֶׁר מִלְּפָנִים הָיִיתִי מְגַדֵּף וּמְרַדֵּף וּמְחָרֵף אֲבָל רֻחַמְתִּי כִּי עָשִׂיתִי מִבְּלִי־דַעַת בְּאֵין אֱמוּנָה׃

יג

13

וַיִּגְדַּל עָלַי בִּמְאׂד מְאׂד חֶסֶד אֲדׂנֵינוּ עִם־הָאֱמוּנָה וְהָאַהֲבָה בַּמָּשִׁיחַ יֵשׁוּעַ׃

יד

14

נֶאֱמָן הַדָּבָר וְרָאוּי לַכּׂל לְהִתְקַבֵּל שֶׁהַמָּשִׁיחַ יֵשׁוּעַ בָּא לָעוֹלָם לְהוֹשִׁיעַ אֶת־הַחַטָּאִים אֲשֶׁר אָנׂכִי הַגָּדוֹל בָּהֶם׃

טו

15

וּבַעֲבוּר זׂאת רֻחַמְתִּי לְמַעַן אֲשֶׁר יַרְאֶה יֵשׁוּעַ הַמָּשִׁיחַ בִּי בָרִאשׁוֹנָה אֶת־כָּל־אׂרֶךְ רוּחוֹ לִהְיוֹתִי לְאוֹת לְכׂל אֲשֶׁר־יָבֹאוּ לְהַאֲמִין בּוֹ לִהְיוֹת לָהֶם חַיֵּי עוֹלָם׃

טז

16

וּמֶלֶךְ עוֹלָמִים הָאֱלׂהִים הַקַּיָּם לָעַד וְהַנֶּעְלָם מֵעַיִן וְהֶחָכָם לְבַדּוֹ לוֹ הַכָּבוֹד וְהֶהָדָר לְעוֹלְמֵי עוֹלָמִים אָמֵן׃

יז

17

אֶת־הַמִּצְוָה הַזּׂאת אֲנִי מְצַוְּךָ בְּנִי טִימוֹתִיּוֹס כְּפִי הַנְּבוּאוֹת הַקּׂדְמוֹת עָלֶיךָ אֲשֶׁר תִּלָּחֵם עַל־יָדָן הַמִּלְחָמָה הַטּוֹבָה׃

יח

18

לֶאֱחׂז בָּאֱמוּנָה וּבְדַעַת נְקִיָּה אֲשֶׁר־יֵשׁ מׂאֲסִים בָּהּ וַתִּשָּׁבֵר אֳנִיַּת אֱמוּנָתָם׃

יט

19

וּמֵהֶם הוּמְנִיּוֹס וַאֲלֶכְסַנְדְּרוֹס אֲשֶׁר מְסַרְתִּים לַשָׂטָן לְמַעַן יִוָּסְרוּ לְבִלְתִּי גַדֵּף עוֹד׃

Chapter 2

וְעַתָּה קׂדֶם כָּל־דָּבָר אֲבַקְשָׁה מִכֶּם לָשֵׂאת תְּפִלּוֹת וְתַחֲנוּנִים וּבַקָּשׁוֹת וְתוֹדוֹת בְּעַד כָּל־בְּנֵי אָדָם׃

א

1

בְּעַד הַמְּלָכִים וְכָל־הַשַּׁלִּיטִים לְמַעַן נִחְיֶה חַיֵּי הַשְׁקֵט וָבֶטַח בְּכָל־חֲסִידוּת וָישֶׁר׃

ב

2

כִּי־כֵן טוֹב וְרָצוּי בְּעֵינֵי אֱלׂהִים מוֹשִׁיעֵנוּ׃

ג

3

אֲשֶׁר חֶפְצוֹ כִּי יִוָּשְׁעוּ כָּל־בְּנֵי הָאָדָם וִיקַבְּלוּ דַּעַת הָאֱמֶת׃

ד

4

כִּי אֶחָד הָאֱלׂהִים וְאֶחָד הַסַּרְסוֹר בֵּין אֱלׂהִים וּבֵין בְּנֵי אָדָם הוּא בֶן־אָדָם הַמָּשִׁיחַ יֵשׁוּעַ׃

ה

5

אֲשֶׁר נָתַן אֶת־נַפְשׁוֹ כּׂפֶר בְּעַד הַכּׂל וְזׂאת הָעֵדוּת הַבָּאָה בְּעִתָּהּ׃

ו

6

אֲשֶׁר אֲנִי הָפְקַדְתִּי לָהּ לְכָרוֹז וּלְשָׁלִיחַ אֱמֶת אֲנִי מַגִּיד בַּמָּשִׁיחַ וְלׂא אֲשַׁקֵּר מוֹרֵה הַגּוֹיִם בֶּאֱמוּנָה וּבֶאֱמֶת׃

ז

7

לָכֵן רְצוֹנִי שֶׁיִּתְפַּלֲלוּ הָאֲנָשִׁים בְּכָל־מָקוֹם וְיִשְׂאוּ יְדֵיהֶם קׂדֶשׁ בְּלִי־רׂגֶז וּמְזִמּוֹת׃

ח

8

וְכֵן גַּם־הַנָּשִׁים תִּתְיַפֶּינָה בְּתִלְבּשֶׁת נָאָה בְּבשֶׁת פָּנִים וּצְנִיעוּת לׂא בְּמַחְלְפוֹת הָרׂאשׁ לׂא בְזָהָב לׂא בִפְנִינִים וְלׂא בְּמַלְבּוּשִׁים יְקָרִים׃

ט

9

אֶלָּא בְּמַעֲשִׂים טוֹבִים כָּרָאוּי לַנָּשִׁים אֲשֶׁר בָּחֲרוּ לָהֶן יִרְאַת אֱלׂהִים׃

י

10

הָאִשָּׁה תִּלְמַד דּוּמָם בְּכָל־הַכְנָעָה׃

יא

11

וְאֵינֶנִּי נׂתֵן רְשׁוּת לָאִשָּׁה לְלַמֵּד אַף לׂא לְהִתְנַשֵׂא עַל־הָאִישׁ אַךְ תִּדּוֹם׃

יב

12

אֲבָל תִּוָּשַׁע בְּלִדְתָּהּ בָּנִים אִם תַּעֲמׂדְנָה בָּאֱמוּנָה וּבָאַהֲבָה וּבַקְּדֻשָּׁה עִם־הַצְּנִיעוּת׃

Chapter 3

אֱמֶת הַדָּבָר אִישׁ כִּי־יְבַקֵּשׁ לִהְיוֹת הֶגְמוֹן לְמַעֲשֶׂה טוֹב מִתְאַוֶּה׃

א

1

וְהֶגְמוֹן הָעֵדָה צָרִיךְ לִהְיוֹת בְּאֵין דּׂפִי בַּעַל־אִשָּׁה אַחַת משֵׁל בְּרוּחוֹ צָנוּעַ וְנֶחְמָד לַבְּרִיּוֹת מַכְנִיס אׂרְחִים וּמֵבִין לְלַמֵּד וְלׂא אׂהֵב יַיִן וְלׂא־בַעַל אֶגְרׂף (וְלׂא בֹצֵעַ בֶּצַע רָע)׃

ב

2

כִּי אִם־דָּן לְכַף־זְכוּת וְלׂא אִישׁ מִדְיָנִים וְלׂא אׂהֵב כָּסֶף׃

ג

3

וִיהִי מַנְהִיג אֶת־בֵּיתוֹ בְּטוֹב וּמַדְרִיךְ אֶת־בָּנָיו לְמִשְׁמַעְתּוֹ בְּכָל־הַיּשֶׁר׃

ד

4

כִּי אִם־לׂא יֵדַע אִישׁ לְהַנְהִיג אֶת־בֵּיתוֹ אֵיכָכָה יוּכַל לְהַשְׁגִּיחַ עַל־עֲדַת אֱלׂהִים׃

ה

5

וְאַל־יְהִי תַּלְמִיד חָדָשׁ פֶּן־יִרְהַב לִבּוֹ וְיִפּׂל בְּדִין הַמַּשְׂטִין׃

ו

6

וְהוּא גַם־צָרִיךְ הֱיוֹת לוֹ שֵׁם טוֹב בְּפִי אׂתָם שֶׁבַּחוּץ פֶּן־יִפּׂל בְּחֶרְפָּה וּבְמוֹקֵשׁ הַמַּשְׂטִין׃

ז

7

וְכֵן גַּם־הַשַּׁמָּשִׁים יִהְיוּ יְשָׁרִים וְלׂא מַחֲלִיקֵי לָשׁוֹן וְלׂא אׂהֲבִים סְבֹא־יַיִן וְלׂא נׂטִים אַחֲרֵי בֶצַע־רָע׃

ח

8

כִּי אִם־יִשְׁמְרוּ אֶת־סוֹד הָאֱמוּנָה בְּרוּחַ טְהוֹרָה׃

ט

9

וְגַם־הֵם יִבָּחֲנוּ בָרִאשׁׂנָה וְאַחַר כֵּן יְשַׁמְּשׁוּ אִם־אֵין בָּהֶם דּׂפִי׃

י

10

וְכֵן הַנָּשִׁים תִּהְיֶינָה יְשָׁרוֹת וְלׂא מַלְשִׁינוֹת משְׁלוֹת בְּרוּחָן וְנֶאֱמָנוֹת בַּכּׂל׃

יא

11

הַשַּׁמָּשִׁים יִהְיוּ כָל־אֶחָד בַּעַל־אִשָּׁה אֶחָת וּמְנַהֲלִים בְּטוֹב אֶת־בְּנֵיהֶם וְאֶת־בָּתֵּיהֶם׃

יב

12

כִּי הַמְשַׁמְּשִׁים כַּהׂגֶן יִקְנוּ לְנַפְשָׁם מַעֲלָה טוֹבָה וּבִטָּחוֹן רַב בֶּאֱמוּנַת הַמָּשִׁיחַ יֵשׁוּעַ׃

יג

13

זׂאת אֲנִי כׂתֵב לְךָ וַאֲקַוֶּה לָבֹא אֵלֶיךָ בִּמְהֵרָה׃

יד

14

וְאִם־אֶתְמַהְמַהּ הִנֵּה תֵדַע אֵיךְ לְהִתְנַהֵג בְּבֵית הָאֱלׂהִים אֲשֶׁר הִיא עֲדַת אֱלׂהִים חַיִּים עַמּוּד הָאֱמֶת וּמְכוֹנָהּ׃

טו

אַל־תִּגְעַר בְּזָקֵן כִּי אִם־תַּזְהִירֶנּוּ כְּאָב־לָךְ וְאֶת־הַצְּעִירִים כְּאַחִים׃

Chapter 6

כּׂל אֲשֶׁר־עׂל הָעַבְדוּת עֲלֵיהֶם יִהְיוּ אֲדׂנֵיהֶם בְּעֵינֵיהֶם רְאוּיִם לְכָל־כָּבוֹד לְמַעַן אֲשֶׁר לׂא־יְחֻלַּל שֵׁם הָאֱלׂהִים וְהַלֶּקַח׃
