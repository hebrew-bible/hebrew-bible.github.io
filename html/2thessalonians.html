2Thessalonians

Chapter 1

פּוֹלוֹס וְסִלְוָנוֹס וְטִימוֹתִיּוֹס אֶל־קְהִלַּת הַתֶּסָּלוֹנִיקִיּים בֵּאלׂהִים אָבִינוּ וּבַאֲדׂנֵינוּ יֵשׁוּעַ הַמָּשִׁיחַ׃

א

1

חֶסֶד לָכֶם וְשָׁלוֹם מֵאֵת אֱלׂהִים אָבִינוּ וַאֲדׂנֵינוּ יֵשׁוּעַ הַמָּשִׁיחַ׃

ב

2

חַיָּבִים אֲנַחְנוּ לְהוֹדוֹת תָּמִיד לֵאלׂהִים עֲלֵיכֶם אַחַי כָּרָאוּי כִּי אֱמוּנַתְכֶם גָּדְלָה מְאׂד וְרַבָּה אַהֲבַת כֻּלְּכֶם אִישׁ לְרֵעֵהוּ׃

ג

3

עַד שֶׁנִּתְהַלֵּל בָּכֶם בִּקְהִלּוֹת אֱלׂהִים עַל־סַבְלָנוּתְכֶם וְעַל־אֱמוּנַתְכֶם בְּכָל־הָרְדִיפוֹת וּבְכָל־הַלַּחַץ אֲשֶׁר סְבַלְתֶּם׃

ד

4

לְאוֹת צִדְקַת מִשְׁפַּט אֱלׂהִים לְמַעַן תִּמָּצְאוּ רְאוּיִם לְמַלְכוּת הָאֱלׂהִים אֲשֶׁר בַּעֲבוּרָהּ תְּעֻנּוּ׃

ה

5

בַּאֲשֶׁר צַדִּיק הוּא הָאֱלׂהִים לִגְמׂל לַחַץ לְלׂחֲצֵיכֶם׃

ו

6

וְלָכֶם הַנִּלְחָצִים רְוָחָה עִמָּנוּ יָחַד בְּהִגָּלוֹת הָאָדוֹן יֵשׁוּעַ מִן־הַשָּׁמַיִם עִם־מַלְאֲכֵי עֻזּוֹ׃

ז

7

בְּאֵשׁ לֶֶהָבָה לְהָשִׁיב נָקָם לַאֲשֶׁר לׂא־יָדְעוּ אֶת־הָאֱלׂהִים וְלׂא שָׁמְעוּ לִבְשׂוֹרַת אֲדׂנֵינוּ יֵשׁוּעַ הַמָּשִׁיחַ׃

ח

8

אֲשֶׁר יִשְׂאוּ מִשְׁפָּטָם אָבְדַּן עוֹלָם מֵאֵת פְּנֵי הָאָדוֹן וּמֵהֲדַר גְּאוֹנוֹ׃

ט

9

בְּבֹאוֹ בַּיּוֹם הַהוּא לְהִכָּבֵד בִּקְדשָׁיו וּלְהִתְפַּלֵּא בְּכָל־הַמַּאֲמִינִים כִּי הֶאֱמַנְתֶּם לְעֵדוּתֵנוּ אֲלֵיכֶם׃

י

10

לָזׂאת גַּם־נִתְפַּלֵּל בַּעַדְכֶם תָּמִיד אֲשֶׁר יִתֵּן אֶתְכֶם אֱלׂהֵינוּ רְאוּיִם לְמַה־שֶּׁאַתֶּם מְקׂרָאִים וִימַלֵּא בְעׂז כָּל־חֶפְצְכֶם בַּטּוֹב וְכָל־מַעֲשֵׂה אֱמוּנַתְכֶם׃

יא

11

לְמַעַן יִכָּבֶד בָּכֶם שֵׁם יֵשׁוּעַ אֲדׂנֵינוּ וְאַתֶּם תִּכָּבְדוּ בוֹ עַל־פִּי חֶסֶד אֱלׂהֵינוּ וַאֲדׂנֵינוּ יֵשׁוּעַ הַמָּשִׁיחַ׃

יב

12

Chapter 2

וְהִנְנוּ מְבַקְשִׁים מִכֶּם אֶחָי עַל־אׂדוֹת בִּיאַת אֲדׂנֵינוּ יֵשׁוּעַ הַמָּשִׁיחַ וַאֲסִיפָתֵנוּ אֵלָיו׃

א

1

אֲשֶׁר לׂא־תִטָּרֵף דַּעְתְּכֶם פִּתְאׂם וַאֲשֶׁר לׂא תִּבָּהֵלוּ לׂא בְרוּחַ וְלׂא בְדִבּוּר וְלׂא בְאִגֶּרֶת כְּאִלּוּ שְׁלוּחָה מֵאִתָּנוּ לֵאמׂר הִגִּיעַ יוֹם יְהוָֹה׃

ב

2

אַל־יַתְעֶה אֶתְכֶם אִישׁ בְּשׁוּם אוֹפֶן כִּי לׂא־יָבֹא בְּטֶרֶם יִהְיֶה בָרִאשׁוֹנָה הַמֶּרֶד וְנִגְלָה אִישׁ הָחַטָּאָה בֶּן־הָאֲבַדּוֹן׃

ג

3

הַמִּתְקוֹמֵם וְהַמִּתְרוֹמֵם עַל־כָּל־הַנִּקְרָא אֵל אוֹ עֲבוֹדָה כִּי גַם־יֵשֵׁב בְּהֵיכַל הָאֱלׂהִים כֵּאלׂהִים וּמַרְאֶה אֶת־עַצְמוֹ כִּי אֱלׂהִים הוּא׃

ד

4

הֲלׂא תִזְכְּרוּ כִּי־בְעוֹד הֱיוֹתִי אִתְּכֶם אֵלֶּה דִּבַּרְתִּי אֲלֵיכֶם׃

ה

5

וְעַתָּה יְדַעְתֶּם אֶת אֲשֶׁר יִמְנָעֵהוּ מִלְּהִגָּלוֹת בְּעִתּוֹ׃

ו

6

כִּי סוֹד הָרֶשַׁע כְּבָר הֵחֵלָּה פְעֻלָּתוֹ רַק שֶׁיּוּסַר מִתָּוֶךְ הַמּׂנֵעַ עַד־כּׂה׃

ז

7

וְאָז יִגָּלֶה הָרָשָׁע אֲשֶׁר יְמִיתֶנּוּ הָאָדוֹן בְּרוּחַ שְׂפָתָיו וְיַשְׁבִּיתֶנּוּ בְּהוֹפָעַת בִּיאָתוֹ׃

ח

8

אֵת אֲשֶׁר יָבוֹא כְּפִי פְּעֻלַּת הַשָׂטָן בְּרָב־כּׂחַ וּבְאׂתוֹת וּבְמוֹפְתֵי שָׁקֶר׃

ט

9

וּבְכָל־מִרְמַת אָוֶן בָּאׂבְדִים תַּחַת אֲשֶׁר לׂא קִבְּלוּ אֶת־אַהֲבַת הָאֱמֶת לְהִוָּשֵׁעַ׃

י

10

וּבַעֲבוּר זׂאת יְשַׁלַּח לָהֶם הָאֱלׂהִים מַדּוּחֵי שָׁוְא לְהַאֲמִין בַּשָּׁקֶר׃

יא

11

לְמַעַן יִדּוֹנוּ כָּל־אֲשֶׁר לׂא־הֶאֱמִינוּ בָאֱמֶת וְרָצוּ בָּעַוְלָה׃

יב

12

אֲבָל אֲנַחְנוּ חַיָּבִים לְהוֹדוֹת תָּמִיד לֵאלׂהִים בַּעֲבוּרְכֶם אַחַי חֲבִיבֵי הָאָדוֹן אֲשֶׁר מִבַּתְּחִלָּה בָּחַר בָּכֶם הָאֱלׂהִים לִישׁוּעָה בִּקְדֻשַּׁת הָרוּחַ וּבֶאֱמוּנַת הָאֱמֶת׃

יג

13

לָזׂאת קָרָא אֶתְכֶם בִּבְשׂוֹרָתֵנוּ לְנַחֲלַת כְּבוֹד אֲדׂנֵינוּ יֵשׁוּעַ הַמָּשִׁיחַ׃

יד

14

לָכֵן אַחַי עִמְדוּ וְהַחֲזִיקוּ בַקַּבָּלוֹת אֲשֶׁר לֻמַּדְתֶּם אִם בִּדְבָרֵנוּ אִם בְּאִגַּרְתֵּנוּ׃

טו

15

וַאֲדׂנֵינוּ יֵשׁוּעַ הַמָּשִׁיחַ וֵאלׂהֵינוּ אָבִינוּ אֲשֶׁר אָהַב אׂתָנוּ וַיִּתֶּן־לָנוּ בְחַסְדּוֹ נֶחָמַת עוֹלָם וְתִקְוָה טוֹבָה׃

טז

16

הוּא יְנַחֵם אֶת־לְבַבְכֶם וִיכוֹנֵן אֶתְכֶם בְּכָל־דָּבָר וּמַעֲשֶׂה טוֹב׃

יז

17

Chapter 3

וְעוֹד אַחַי הִתְפַּלֲלוּ בַעֲדֵנוּ אֲשֶׁר יָרוּץ דְּבַר יְהוָֹה וְיִכָּבֵד כְּהִכָּבְדוֹ בְקִרְבְּכֶם׃

א

1

וַאֲשֶׁר נִנָּצֵל מִן־הָאֲנָשִׁים הַתּׂעִים וְהָרָעִים כִּי לׂא לְכָל־אָדָם הָאֱמוּנָה׃

ב

2

אֲבָל נֶאֱמָן הוּא הָאָדוֹן אֲשֶׁר יְחַזֵּק אֶתְכֶם וְיִשְׁמָרְכֶם מִן־הָרָע׃

ג

3

וַאֲנַחְנוּ בְּטוּחִים עֲלֵיכֶם בַּאֲדׂנֵינוּ שֶׁעׂשִׂים אַתֶּם כְּמִצְוָתֵנוּ וְגַם־תּוֹסִיפוּ לַעֲשׂוֹת׃

ד

4

וְהָאָדוֹן הוּא יְיַשֵּׁר אֶת־לְבַבְכֶם לְאַהֲבַת הָאֱלׂהִים וּלְסַבְלָנוּת הַמָּשִׁיחַ׃

ה

5

וְהִנְנוּ מְצַוִּים אֶתְכֶם אַחִים בְּשֵׁם אֲדׂנֵינוּ יֵשׁוּעַ הַמָּשִׁיחַ אֲשֶׁר תִּבָּדְלוּ מִכָּל־אָח מְעַקֵּשׁ דְּרָכָיו וְאֵינֶנּוּ מִתְהַלֵּךְ עַל־פִּי הַקַּבָּלָה אֲשֶׁר קִבֵּל מֵאִתָּנוּ׃

ו

6

הֲלׂא אַתֶּם יְדַעְתֶּם שֶׁצְּרִיכִים אַתֶּם לְהִתְהַלֵּךְ כָּמׂנוּ כִּי לׂא־הֶעֱוִינוּ אֶת־דַּרְכֵּנוּ בְּתוֹכֲכֶם׃

ז

7

גַּם לׂא־אָכַלְנוּ לֶחֶם אִישׁ חִנָּם כִּי בְּעָמָל וּתְלָאָה לַיְלָה וְיוֹמָם יָגַעְנוּ לְבִלְתִּי הֱיוֹת לְאִישׁ מִכֶּם לְמַשָׂא׃

ח

8

לׂא מִפְּנֵי שֶׁאֵין־לָנוּ הָרְשׁוּת הַזּׂאת כִּי אִם־לָתֵת אׂתָנוּ לָכֶם לְמוֹפֵת לָלֶכֶת בְּעִקְּבוֹתֵינוּ׃

ט

9

כִּי גַּם־בִּהְיוֹתֵנוּ אֶצְלְכֶם צִוִּינוּ אֶתְכֶם לֵאמׂר אֲשֶׁר לׂא יִרְצֶה לַעֲשׂוֹת גַּם־לׂא יׂאכֵל׃

י

10

כִּי שָׁמַעְנוּ שֶׁיֵּשׁ בָּכֶם מְעַקְּשִׁים אָרְחׂתָם וְאֵינָם עׂשִׂים דָּבָר וּמְבַלִּים יְמֵיהֶם בַּהֲבָלִים׃

יא

11

וַאֲנָשִׁים כָּאֵלֶּה נְצַוֶּה אוֹתָם וְנַזְהִירֵם בַּאֲדׂנֵינוּ יֵשׁוּעַ הַמָּשִׁיחַ לַעֲבֹד בְּנַחַת לְמַעַן יׂאכְלוּ אֶת־לַחְמָם מִשֶּׁלָּהֶם׃

יב

12

וְאַתֶּם אֶחָי אַל־תִּתְרַפּוּ בַּעֲשׂוֹת הַטּוֹב׃

יג

13

וְאִם לׂא־יִשְׁמַע אִישׁ אֶל־דְּבָרֵנוּ בָּאִגֶּרֶת הַזּׂאת אׂתוֹ תִרְשְׁמוּ לָכֶם וְאַל־תִּתְעָרֲבוּ עִמּוֹ לְמַעַן יֵבוֹשׁ׃

יד

14

אַךְ לׂא־כְאׂיֵב תַּחְשְׁבֻהוּ כִּי אִם־תּוֹכִיחֻהוּ כְּאָח׃

טו

15

וַאֲדוֹן הַשָּׁלוֹם הוּא יִתֵּן לָכֶם הַשָּׁלוֹם תָּמִיד וּבְכָל־פָּנִים׃

טז

16

יְהִי הָאָדוֹן עִם־כֻּלְּכֶם׃

יז

17

שׁׂאֵל לִשְׁלוֹמְכֶם בִּכְתָב יָדִי אֲנִי פוֹלוֹס וְהִיא הָאוֹת בְּכָל־הָאִגְּרוֹת כֵּן אָנׂכִי כׂתֵב׃

יח

18
