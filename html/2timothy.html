Chapter 1

פּוֹלוֹס שְׁלִיחַ יֵשׁוּעַ הַמָּשִׁיחַ בִּרְצוֹן הָאֱלׂהִים לְפִי הַבְטָחַת הַחַיִּים אֲשֶׁר בְּיֵשׁוּעַ הַמָּשִׁיחַ׃

א

1

אֶל־טִימוֹתִיּוֹס בְּנִי הֶחָבִיב חֶסֶד וְרַחֲמִים וְשָׁלוֹם מֵאֵת אֱלׂהִים אָבִינוּ וּמֵאֵת הַמָּשִׁיחַ יֵשׁוּעַ אֲדׂנֵינוּ׃

ב

2

מוֹדֶה אֲנִי לֵאלׂהִים אֲשֶׁר אָנׂכִי עׂבֵד אוֹתוֹ מִימֵי אֲבוֹתַי בְּרוּחַ טְהוֹרָה כִּי־תָמִיד אֶזְכָּרְךָ בִּתְפִלּוֹתַי לַיְלָה וְיוֹמָם׃

ג

3

וְנִכְסַפְתִּי לִרְאוֹתְךָ בְּזָכְרִי אֶת־דִּמְעוֹתֶיךָ לְמַעַן אֶמָּלֵא שִׂמְחָה׃

ד

4

כִּי־שַׂמְתִּי עַל לִבִּי זִכְרוֹן אֱמוּנָתְךָ בִּלְתִּי־צְבוּעָה שֶׁכְּבָר הָיְתָה בְּלוֹאִיס זְקֵנָתְךָ וּבְאַבְנִיקִי אִמֶּךָ וּמֻבְטָח אֲנִי שֶׁהִיא גַּם־בָּךְ׃

ה

5

וּבַעֲבוּר זׂאת אַזְכִּירְךָ כִּי תְעוֹרֵר אֶת־מַתְּנַת הָאֱלׂהִים אֲשֶׁר בְּךָ בִּסְמִיכַת יָדִי׃

ו

6

כִּי הָאֱלׂהִים לׂא־נָתַן לָנוּ רוּחַ אֵימָה כִּי אִם־רוּחַ גְּבוּרָה וְאַהֲבָה וּמוּסָר׃

ז

7

לָכֵן אַל־תֵּבוֹשׁ לׂא מֵעֵדוּת אֲדׂנֵינוּ וְלׂא מִמֶּנִּי אֲסִירוֹ כִּי אִם־תְּקַבֵּל כָּמוֹנִי הָרָעוֹת עַל־הַבְּשׂוֹרָה בִּגְבוּרַת אֱלׂהִים׃

ח

8

אֲשֶׁר־הוֹשִׁיעָנוּ וּקְרָאָנוּ קְרִיאָה קְדוֹשָׁה לׂא כְמַעֲשֵׂינוּ כִּי אִם־כַּעֲצָתוֹ וְחַסְדּוֹ הַנִּתָּן לָנוּ בַּמָּשִׁיחַ יֵשׁוּעַ לִפְנֵי יְמוֹת עוֹלָם׃

ט

9

וְעַתָּה נִגְלָה בְּהֵרָאוֹת מוֹשִׁיעֵנוּ יֵשׁוּעַ הַמָּשִׁיחַ אֲשֶׁר בִּלַּע הַמָּוֶת וַיּוֹצֵא לָאוֹר אֶת־הַחַיִּים וְאֵת אֲשֶׁר־אֵינֶנּוּ עוֹבֵר עַל־יְדֵי הַבְּשׂוֹרָה׃

י

10

אֲשֶׁר הָפְקַדְתִּי לִהְיוֹת לָהּ לְכָרוֹז וְשָׁלִיחַ וּמוֹרֵה הַגּוֹיִם׃

יא

11

וּבַעֲבוּר זׂאת אֶסְבֹּל כָּאֵלֶּה וְלׂא אֵבוֹשׁ כִּי־יָדַעְתִּי בְּמִי הֶאֱמַנְתִּי וַאֲנִי מֻבְטָח שֶׁבְּיָדוֹ לִשְׁמׂר אֶת־פִּקְדוֹנִי לַיּוֹם הַהוּא׃

יב

12

הַחֲזֵקּ בְּתַבְנִית הַדְּבָרִים הַבְּרִיאִים אֲשֶׁר שָׁמַעְתָּ מִמֶּנִּי בָּאֱמוּנָה וּבָאַהֲבָה אֲשֶׁר בַּמָּשִׁיחַ יֵשׁוּעַ׃

יג

13

שְׁמׂר אֶת־הַפִּקָּדוֹן הַטּוֹב עַל־יְדֵי רוּחַ הַקּׂדֶשׁ הַשּׁׂכֵן בָּנוּ׃

יד

14

זׂאת אַתָּה יוֹדֵעַ כִּי־פָנוּ מִמֶּנִּי כּׂל אֲשֶׁר בְּאַסְיָא וּבְתוֹכָם פּוּגְלוֹס וְהַרְמוֹגְנִיס׃

טו

15

יִתֵּן הָאָדוֹן רַחֲמִים לְבֵית אֳנִיסִיפוֹרוֹס כִּי־פְעָמִים רַבּוֹת הֵשִׁיב אֶת־נַפְשִׁי וְלׂא־בוֹשׁ מִמּוֹסֵרָי׃

טז

16

כִּי אִם־בִּהְיוֹתוֹ בְרוֹמִי יָגַע לְבַקְּשֵׁנִי עַד כִּי־מְצָאָנִי׃

יז

17

יִתְּנֶנּוּ הָאָדוֹן לְרַחֲמִים לִפְנֵי יְהוָָֹה בַּיּוֹם הַהוּא וְרׂב שֵׁרוּתוֹ אֲשֶׁר שֵׁרֵת בְּאֶפְסוֹס אַתָּה יָדַעְתָּ הֵיטֵב׃

Chapter 2

וְעַתָּה בְּנִי הִתְחַזֵּק בַּחֶסֶד אֲשֶׁר בַּמָּשִׁיחַ יֵשׁוּעַ׃

א

1

וְאֶת־אֲשֶׁר שָׁמַעְתָּ מִמֶּנִּי בִּפְנֵי עֵדִים רַבִּים תַּפְקִידֶנּוּ בִּידֵי אֲנָשִׁים נֶאֱמָנִים אֲשֶׁר־הֵם כְּשֵׁרִים לְלַמֵּד גַּם אֶת־הָאֲחֵרִים׃

ב

2

וְקַבֵּל הָרָעוֹת כְּגִבּוֹר חַיִל בְּמִלְחֶמֶת יֵשׁוּעַ הַמָּשִׁיחַ׃

ג

3

אִישׁ יׂצֵא לַצָּבָא לׂא יִתְעָרֵב בְּעִסְקֵי הַחַיִּים לְמַעַן יִהְיֶה רָצוּי לְשַׂר הַצָּבָא׃

ד

4

וְגַּם אִם־נֶאֱבַק אִישׁ לׂא יֻכְתַּר אִם־לׂא יֵאָבֵק כַּמִּשְׁפָּט׃

ה

5

הָאִכָּר הָעׂבֵד הוּא יׂאכֵל רִאשׁוֹנָה מִפְּרִי הָאֲדָמָה׃

ו

6

בִּין בַּאֲשֶׁר־אֲנִי אׂמֵר כִּי הָאָדוֹן יִתֶּן־לְךָ חָכְמָה בְּכָל־דָּבָר׃

ז

7

זָכוֹר תִּזְכּׂר אֵת יֵשׁוּעַ הַמָּשִׁיחַ הַנֵּעוֹר מִן־הַמֵּתִים אֲשֶׁר הוּא מִזֶּרַע דָּוִד כִּבְשׂוֹרָתִי׃

ח

8

אֲשֶׁר בַּעֲבוּרָהּ אֲנִי נׂשֵׂא רָעוֹת עַד לַמּוֹסֵרוֹת כְּעׂשֶׂה עָוֶל אֲבָל דְּבַר הָאֱלׂהִים אֵינֶנּוּ חָבוּשׁ׃

ט

9

וְעַל־כֵּן אֶסְבֹּל אֶת־כּׂל לְמַעַן הַבְּחִירִים לְמַעַן יַשִׂיגוּ גַם־הֵמָּה אֶת־הַתְּשׁוּעָה בַּמָּשִׁיחַ יֵשׁוּעַ עִם־כְּבוֹד עוֹלָמִים׃

י

10

אֱמֶת הַדָּבָר הַזֶּה כִּי אִם־מַתְנוּ אִתּוֹ גַּם־אִתּוֹ נִחְיֶה׃

יא

11

אִם־נִסְבֹּל גַּם־נִמְלׂךְ אִתּוֹ וְאִם־נְכַחֶשׁ־בּוֹ גַּם־הוּא יְכַחֶשׁ־בָּנוּ׃

יב

12

אִם־לׂא נַאֲמִין הִנֵּה הוּא קַיָּם בֶּאֱמוּנָתוֹ כִּי לׂא יוּכַל לְכַחֵשׁ בְּעַצְמוֹ׃

יג

13

זׂאת הַזְכֵּר לָהֶם וְהָעֵד לִפְנֵי הָאָדוֹן שֶׁלּא לַעֲסׂק בְּתוֹכַחַת מִלִּים לׂא לְהוֹעִיל רַק לְעַוֵּת דַּעַת הַשּׁׂמְעִים׃

יד

14

הֱיֵה שָׁקוּד לְהִתְיַצֵּב נֶאֱמָן לִפְנֵי אֱלׂהִים וּכְפׂעֵל אֲשֶׁר לׂא־יֵבוֹשׁ הַמְחַלֵּק עַל־נָכוֹן דְּבַר הָאֱמֶת׃

טו

15

אֲבָל תִּרְחַק מִדִּבְרֵי הֶבֶל הַפְּסוּלִים כִּי יוֹסִיפוּ הַרְבּוֹת רֶשַׁע׃

טז

16

וְשִׂיחָתָם כְּרָקָב תּׂאכַל סָבִיב וּמֵהֶם הוּמְנִיּוֹס וּפִילִיטוֹס׃

יז

17

אֲשֶׁר תָּעוּ מִן־הָאֱמֶת בְּאָמְרָם כְּבָר הָיְתָה תְּחִיַּת הַמֵּתִים וַיְבַלְבְּלוּ אֱמוּנַת אֲחָדִים׃

יח

18

וִיסוֹד הָאֱלׂהִים אֵיתָן לׂא יִמּוֹט וְזֶה חוֹתָמוֹ יׂדֵעַ יְהוָֹה אֵת אֲשֶׁר־לוֹ וְכָל־הַקּוֹרֵא בְּשֵׁם יְהוָֹה יָסוּר מֵעָוֶל׃

יט

19

וּבְבַיִת גָּדוֹל לׂא־כְלֵי כֶסֶף וְזָהָב בִּלְבַד אֶלָּא גַם־שֶׁל־עֵץ וְשֶׁל־חֶרֶשׂ וּמֵהֶם לְכָבוֹד וּמֵהֶם לְבִזָּיוֹן׃

כ

20

וְהִנֵּה אִם־טִהַר אִישׁ אֶת־נַפְשׁוֹ מֵאֵלֶּה יִהְיֶה כְלִי לְכָבוֹד מְקֻדָּשׁ וּמוֹעִיל לָאָדוֹן מוּכָן לְכָל־מַעֲשֶׂה טוֹב׃

כא

21

בְּרַח־לְךָ מִתַּאֲוֹת הַנְּעוּרִים וּרְדׂף צֶדֶק וֶאֱמוּנָה וְאַהֲבָה וְשָׁלוֹם עִם־כּׂל הַקּׂרְאִים אֶל־אֲדׂנָי בְּלֵב טָהוֹר׃

כב

22

וְהַרְחֵק מִן־הַשְּׁאֵלוֹת הַתְּפֵלוֹת בְּאֵין מוּסָר מִדַּעְתְּךָ כִּי־אַךְ מְרִיבוֹת מוֹלִידוֹת הֵנָּה׃

כג

23

וְעֶבֶד הָאָדוֹן לׂא יָרִיב אַךְ יְהִי־נוֹחַ לַכּׂל וּמֵבִין לְלַמֵּד וְסַבְלָן׃

כד

24

וּמְיַסֵּר בַּעֲנָוָה אֶת־הַמַּכְחִישִׁים אוּלַי יִתֵּן הָאֱלׂהִים בִּלְבָבָם לָשׁוּב וְלָדַעַת הָאֱמֶת׃

כה

25

וְיָעִירוּ מֵרֶשֶׁת הַשָׂטָן אֲשֶׁר צָדָם בָּהּ לִרְצוֹנוֹ׃

Chapter 3

וְזׂאת תֵּדַע כִּי בְּאַחֲרִית הַיָּמִים יָבֹאוּ עִתִּים קָשׁוֹת׃

א

1

כִּי־יִהְיוּ הָאֲנָשִׁים אׂהֲבֵי עַצְמָם וְאׂהֲבֵי בֶצַע וּמִתְהַלֲלִים וְגֵאִים וּמְגַדְּפִים וּמַמְרִים בַּאֲבוֹתָם וּכְפוּיֵי טוֹבָה וְלׂא חֲסִידִים׃

ב

2

חַסְרֵי אַהֲבָה וּבוֹגְדִים וּמַלְשִׁינִים וְזוֹלְלִים וְאַכְזָרִים וְשׂנְאֵי טוֹב׃

ג

3

וּמׂסְרִים וּפׂחֲזִים וּגְבֹהֵי רוּחַ וְאׂהֲבֵי עֲדָנִים וְלׂא אֱלׂהִים׃

ד

4

וַאֲשֶׁר דְּמוּת חֲסִידוּת לָהֶם וּמְכַחֲשִׁים בְּכׂחָהּ וְאַתָּה סוּר מֵאֵלֶּה׃

ה

5

כִּי־מֵהֶם הַמִּתְגַּנְּבִים אֶל־הַבָּתִּים וְשׁׂבִים נְעָרוֹת טְעוּנוֹת חֲטָאִים וְנִתְעוֹת בְּתַאֲוֹת שׁׂנוֹת׃

ו

6

הַלּמְדוֹת תָּמִיד וּלְעוֹלָם אֵינָן יְכׂלוֹת לָבוֹא לִידִיעַת הָאֱמֶת׃

ז

7

וּכְמוֹ יַנִּיס וְיַמְבְּרִיס אֲשֶׁר־קָמוּ עַל־משֶׁה כֵּן גַּם־אֵלֶּה מִתְקוֹמֲמִים אֶל־הָאֱמֶת אֲנָשִׁים אֲשֶׁר נִשְׁחֲתָה דַעְתָּם וּנְלוֹזִים בָּאֱמוּנָה׃

ח

8

אֲבָל לׂא־יוֹסִיפוּ לְהַצְלִיחַ כִּי שִׁגְעוֹנָם יִגָּלֶה לַכּׂל כַּאֲשֶׁר קָרָה גַּם־אֶת־הָאֲנָשִׁים הָהֵם׃
