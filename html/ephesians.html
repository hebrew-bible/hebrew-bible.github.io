Ephesians

Chapter 1

פּוֹלוֹס שְׁלִיחַ יֵשׁוּעַ הַמָּשִׁיחַ בִּרְצוֹן אֱלׂהִים אֶל־הַקְּדשִׁים הַנִּמְצְאִים (בְּאֶפְסוֹס) וּמַאֲמִינִים בַּמָּשִׁיחַ יֵשׁוּעַ׃

א

1

חֶסֶד לָכֶם וְשָׁלוֹם מֵאֵת הָאֱלׂהִים אָבִינוּ וַאֲדׂנֵינוּ יֵשׁוּעַ הַמָּשִׁיחַ׃

ב

2

בָּרוּךְ הָאֱלׂהִים וַאֲבִי אֲדׂנֵינוּ יֵשׁוּעַ הַמָּשִׁיחַ אֲשֶׁר בֵּרְכָנוּ בְּכָל־בִּרְכַּת רוּחַ בַּמְּרוֹמִים בַּמָּשִׁיחַ׃

ג

3

כַּאֲשֶׁר בָּחַר אׂתָנוּ בוֹ לִפְנֵי מוֹסְדוֹת תֵּבֵל לִהְיוֹת קְדשִׁים וּתְמִימִים לְפָנָיו בְּאַהֲבָה׃

ד

4

יְעָדָנוּ לוֹ לְבָנִים עַל־יְדֵי יֵשׁוּעַ הַמָּשִׁיחַ כִּרְצוֹן חֶפְצוֹ׃

ה

5

לִתְהִלַּת כְּבוֹד חַסְדּוֹ אֲשֶׁר־חָנַן אׂתָנוּ בִּידִידוֹ׃

ו

6

אֲשֶׁר־בּוֹ לָנוּ הַפִּדְיוֹם בְּדָמוֹ וּסְלִיחַת הַפְּשָׁעִים כְּרׂב חַסְדּוֹ׃

ז

7

אֲשֶׁר בָּא עָלֵינוּ בְּשֶׁפַע־חָכְמָה וְהַשְׂכֵּל׃

ח

8

וְהוֹדִיעַ אׂתָנוּ אֶת־סוֹד חֶפְצוֹ כִּרְצוֹנוֹ וַעֲצָתוֹ מֵרׂאשׁ אֲשֶׁר יָעַץ בּוֹ׃

ט

9

עַל־דְּבַר הַנְהָגָתוֹ בִּמְלׂאת הַיָּמִים לְקַבֵּל אֶת־הַכּׂל בַּמָּשִׁיחַ מַה־שֶּׁבַּשָּׁמַיִם וּמַה־שֶּׁבָּאָרֶץ׃

י

10

וּבוֹ לָקַחְנוּ נַחֲלָתֵנוּ אֲנַחְנוּ הַמְיֻעָדִים לָהּ מִלְּפָנִים בְּמַחֲשֶׁבֶת פּׂעֵל הַכּׂל כַּעֲצַת חֶפְצוֹ׃

יא

11

לִהְיוֹת לִתְהִלַּת כְּבוֹדוֹ אֲנַחְנוּ אֲשֶׁר יִחַלְנוּ אֶל־הַמָּשִׁיחַ מֵאָז׃

יב

12

וַאֲשֶׁר גַּם־אַתֶּם שְׁתוּלִים בּוֹ אַחֲרֵי שָׁמְעֲכֶם דְּבַר הָאֱמֶת אֶת־בְּשׂוֹרַת יְשׁוּעַתְכֶם וַאֲשֶׁר בּוֹ כְּשֶׁהֶאֱמַנְתֶּם גַּם־נֶחְתַּמְתֶּם בְּרוּחַ הַהַבְטָחָה רוּחַ הַקּׂדֶשׁ׃

יג

13

כִּי זֶה עֶרְבוֹן נַחֲלָתֵנוּ לִפְדּוֹת־לוֹ עַם סְגֻלָּה לִתְהִלַּת כְּבוֹדוֹ׃

יד

14

בַּעֲבוּר זׂאת גַּם־אָנׂכִי אַחֲרֵי שָׁמְעִי אֱמוּנַתְכֶם בַּאֲדׂנֵינוּ יֵשׁוּעַ וְהָאַהֲבָה אֲשֶׁר־אֲהַבְתֶּם אֶת־כָּל־הַקְּדשִׁים׃

טו

15

לׂא אֶחְדַּל מֵהוֹדוֹת בַּעֲבוּרְכֶם בְּהַזְכִּירִי אֶתְכֶם בִּתְפִלּתַי׃

טז

16

שֶׁיִּתֵּן לָכֶם אֱלׂהֵי אֲדׂנֵינוּ יֵשׁוּעַ הַמָּשִׁיחַ אֲבִי הַכָּבוֹד אֶת־רוּחַ הַחָכְמָה וְהֶחָזוֹן לָדַעַת אׂתוֹ׃

יז

17

וְיָאֵר עֵינֵי לְבַבְכֶם לְמַעַן תֵּדְעוּ מַה־הִיא תּוֹחֶלֶת קְרִיאָתוֹ וּמָה־עשֶׁר כְּבוֹד נַחֲלָתוֹ בַּקְּדשִׁים׃

יח

18

וּמַה־יִּתְרוֹן גְּדֻלַּת גְּבוּרָתוֹ בָּנוּ הַמַּאֲמִינִים כְּפִי פְעֻלַּת עׂצֶם כּׂחוֹ׃

יט

19

אֲשֶׁר פָּעַל בַּמָּשִׁיחַ בְּהָעִיר אׂתוֹ מִן־הַמֵּתִים וַיּוֹשִׁיבֶנּוּ לִימִינוֹ בַּמְּרוֹמִים׃

כ

20

מִמַּעַל לְכָל־שְׂרָרָה וְשָׁלְטָן וּגְבוּרָה וּמֶמְשָׁלָה וְכׂל הַנִּקְרָא בְשֵׁם לׂא־בָעוֹלָם הַזֶּה בִּלְבַד כִּי אִם־גַּם בָּעוֹלָם הַבָּא׃

כא

21

וַיָּשֶׁת כּׂל תַּחַת רַגְלָיו וַיִּתֵּן אׂתוֹ לְרׂאשׁ לְכׂל אֶל־הָעֵדָה׃

כב

22

אֲשֶׁר הִיא גּוּפוֹ מְלוֹא שֶׁל־הַמְמַלֵּא אֶת־הַכּׂל בַּכּׂל׃

כג

23

Chapter 2

גַּם־אֶתְכֶם אֲשֶׁר מֵתִים הֱיִיתֶם בְּפִשְׁעֵיכֶם וְחַטּׂאתֵיכֶם׃

א

1

אֲשֶׁר הִתְהַלַּכְתֶּם בָּהֶם כְּדוֹר הָעוֹלָם הַזֶּה כִּרְצוֹן שַׂר מֶמְשֶׁלֶת הָאֲוִיר וְהוּא הָרוּחַ הַפּׂעֵל כָּעֵת בִּבְנֵי הַמֶּרִי׃

ב

2

וְגַם־אֲנַחְנוּ כֻלָּנוּ בְּתוֹכָם הָלַכְנוּ לְפָנִים בְּתַאֲוֹת הַבָּשָׂר לַעֲשׂוֹת חֶפְצֵי בְשָׂרֵנוּ וּמַחְשְׁבוֹתֵינוּ וַנְּהִי בְנֵי־רׂגֶז בְּטִבְעֵנוּ כִּשְׁאָר בְּנֵי אָדָם׃

ג

3

אֲבָל־הָאֱלׂהִים הַמָּלֵא רַחֲמִים בְּרׂב אַהֲבָתוֹ אֲשֶׁר אָהַב אׂתָנוּ׃

ד

4

אַחֲרֵי הֱיוֹתֵנוּ מֵתִים בַּפְּשָׁעִים הֶחֱיָנוּ עִם־הַמָּשִׁיחַ בַּחֶסֶד נוֹשַׁעְתֶּם׃

ה

5

וַיְעִירֵנוּ אִתּוֹ אַף־הוֹשִׁיבָנוּ בַּמְּרוֹמִים בַּמָּשִׁיחַ יֵשׁוּעַ׃

ו

6

לְהַרְאוֹת בַּדּׂרוֹת הַבָּאִים אֶת־גְּדֻלַּת עשֶׁר חַסְדּוֹ בְּטוֹבָתוֹ עָלֵינוּ בַּמָּשִׁיחַ יֵשׁוּעַ׃

ז

7

כִּי־בַחֶסֶד נוֹשַׁעְתֶּם עַל־יְדֵי הָאֱמוּנָה וְלׂא מִיֶּדְכֶם הָיְתָה זּׂאת כִּי־מַתַּת אֱלׂהִים הִיא׃

ח

8

לׂא מִתּוֹךְ הַמַּעֲשִׂים שֶׁלּא יִתְהַלֵּל אִישׁ׃

ט

9

כִּי־פׂעַל אֱלׂהִים אֲנַחְנוּ נִבְרָאִים בַּמָּשִׁיחַ יֵשׁוּעַ לְמַעֲשִׂים טוֹבִים אֲשֶׁר הֵכִין הָאֱלׂהִים מִקֶּדֶם לְמַעַן נִתְהַלֵּךְ בָּהֶם׃

י

10

עַל־כֵּן זִכְרוּ כִּי־אַתֶּם מִתְּחִלָּה הַגּוֹיִם בַּבָּשָׂר הַנִּקְרָאִים עֲרֵלִים בְּפִי הַנִּקְרָאִים בְּנֵי הַמִּילָה שֶׁהִיא מַעֲשֵׂה יָדַיִם בַּבָּשָׂר׃

יא

11

כִּי־אַתֶּם בָּעֵת הַהִיא הֱיִיתֶם בְּלִי מָשִׁיחַ זָרִים מֵעֲדַת יִשְׂרָאֵל וְנָכְרִים לִבְרִיתוֹת הַהַבְטָחָה בְּאֵין תִּקְוָה וּבְאֵין לָכֶם אֱלׂהִים בָּעוֹלָם׃

יב

12

וְעַתָּה בְּיֵשׁוּעַ הַמָּשִׁיחַ אַתֶּם הָרְחוֹקִים מֵאָז הֱיִיתֶם קְרוֹבִים בְּדַם הַמָּשִׁיחַ׃

יג

13

כִּי הוּא שְׁלוֹמֵנוּ אֲשֶׁר עָשָׂה הַשְּׁנַיִם אֶחָד וְהָרַס מְחִיצַת הַגָּדֵר׃

יד

14

בְּבַטְּלוֹ הָאֵיבָה בִּבְשָׂרוֹ אֶת־תּוֹרַת הַמִּצְוֹת וְהַחֻקּוֹת לִבְרׂא בְנַפְשׁוֹ אֶת־הַשְּׁנַיִם לְאָדָם אֶחָד חָדָשׁ וַיַּעַשׂ שָׁלוֹם׃

טו

15

וַיְרַצֶּה אֶת־שְׁנֵיהֶם בְּגוּף אֶחָד לֵאלׂהִים עַל־יְדֵי צְלִיבָתוֹ בַּהֲמִיתוֹ בְנַפְשׁוֹ אֶת־הָאֵיבָה׃

טז

16

וַיָּבֹא וַיְבַשֵׂר לָכֶם שָׁלוֹם שָׁלוֹם לָרָחוֹק וְלַקָּרוֹב׃

יז

17

כִּי עַל־יָדוֹ יֵשׁ לִשְׁנֵינוּ מָבוֹא בְּרוּחַ אֶחָד אֶל־אָבִינוּ׃

יח

18

לָכֵן אֵינְכֶם עוֹד גֵּרִים וְתוֹשָׁבִים כִּי־בְנֵי־עִירָם שֶׁל־הַקְּדשִׁים וּבְנֵי בֵּית אֱלׂהִים אַתֶּם׃

יט

19

בְּנוּיִם עַל־יְסוֹד הַשְּׁלִיחִים וְהַנְּבִיאִים וְיֵשׁוּעַ הַמָּשִׁיחַ הוּא אֶבֶן הַפִּנָּה׃

כ

20

אֲשֶׁר חֻבַּר־בּוֹ יַחַד הַבִּנְיָן כֻּלּוֹ עַד אֲשֶׁר־יִגְבַּהּ לְהֵיכַל קׂדֶשׁ לַיהוָֹה׃

כא

21

וּבוֹ נִבְנִים גַּם־אַתֶּם לִהְיוֹת מִשְׁכַּן אֱלׂהִים בָּרוּחַ׃

כב

22

Chapter 3

בַּעֲבוּר זׂאת אֲנִי פוֹלוֹס אֲסִיר הַמָּשִׁיחַ לְמַעַנְכֶם הַגּוֹיִם׃

א

1

כִּי־אָמְנָם שְׁמַעְתֶּם הַנְהָגַת חֶסֶד הָאֱלׂהִים אֲשֶׁר־נִתַּן לִי אֲלֵיכֶם׃

ב

2

כִּי בֶחָזוֹן גִּלָּה־לִי הַסּוֹד אֲשֶׁר כָּתַבְתִּי לָכֶם בִּמְעַט מִלִּין׃

ג

3

וּבְקָרְאֲכֶם תּוּכְלוּ לְהַכִּיר אֶת־בִּינָתִי בְּסוֹד הַמָּשִׁיחַ׃

ד

4

אֲשֶׁר בַּדּוֹרׂת הָרִאשׁׂנִים לׂא נוֹדַע לִבְנֵי אָדָם כַּאֲשֶׁר נִגְלָה עַתָּה לִשְׁלִיחָיו הַקְּדשִׁים וְלִנְבִיאָיו בָּרוּחַ׃

ה

5

שֶׁיִּהְיוּ הַגּוֹיִם גַּם־הֵם נׂחֲלִים וּבְגוּף אֶחָד וְחַבְרֵי הַהַבְטָחָה בַּמָּשִׁיחַ יֵשׁוּעַ עַל־יְדֵי הַבְּשׂרָה׃

ו

6

אֲשֶׁר הָיִיתִי לָהּ לִמְשָׁרֵת בְּמַתְּנַת חֶסֶד אֱלׂהִים הַנְּתוּנָה לִי כְּפִי פְּעֻלַּת גְּבוּרָתוֹ׃

ז

7

לִי אֲנִי צְעִיר הַצְּעִירִים שֶׁבְּכָל־הַקְּדשִׁים נִתַּן הַחֶסֶד הַזֶּה לְבַשֵׂר בַּגּוֹיִם אֶת־עשֶׁר הַמָּשִׁיחַ אֲשֶׁר רַב הוּא לְאֵין חֵקֶר׃

ח

8

וּלְהָאִיר עֵינֵי כׂל לִרְאוֹת מַה־הַנְהָגַת הַסּוֹד הַנִּסְתָּר מִדּׂרׂת עוֹלָם בֵּאלׂהִים יוֹצֵר הַכּׂל (עַל־יְדֵי יֵשׁוּעַ הַמָּשִׁיחַ)׃

ט

9

לְמַעַן אֲשֶׁר תִּוָּדַע עַתָּה עַל־יְדֵי הָעֵדָה לַשָׂרִים וְלַשַּׁלִּיטִים אֲשֶׁר בַּמְּרוֹמִים חָכְמַת אֱלׂהִים הַמַּפְלִיאָה דְּרָכֶיהָ׃
