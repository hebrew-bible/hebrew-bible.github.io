Galatians

Chapter 1

פּוֹלוֹס הַשָּׁלִיחַ לׂא מִבְּנֵי אָדָם וְלׂא עַל־יְדֵי בֶן־אָדָם כִּי אִם־עַל־יְדֵי יֵשׁוּעַ הַמָּשִׁיחַ וֵאלׂהִים הָאָב אֲשֶׁר הֱעִירוֹ מִן־הַמֵּתִים׃

א

1

וְכָל־הָאַחִים אֲשֶׁר עִמָּדִי אֶל־הַקְּהִלּוֹת אֲשֶׁר בְּגָלַטְיָא׃

ב

2

חֶסֶד לָכֶם וְשָׁלוֹם מֵאֵת הָאֱלׂהִים אָבִינוּ וּמֵאֵת אֲדׂנֵינוּ יֵשׁוּעַ הַמָּשִׁיחַ׃

ג

3

אֲשֶׁר־נָתַן אֶת־נַפְשׁוֹ עַל־חַטּׂאתֵינוּ לְחַלְּצֵנוּ מִן־הָעוֹלָם הָרָע הַזֶּה כִּרְצוֹן אֱלׂהֵינוּ אָבִינוּ׃

ד

4

אֲשֶׁר־לוֹ הַכָּבוֹד לְעוֹלְמֵי עוֹלָמִים אָמֵן׃

ה

5

תָּמֵהַּ אֲנִי כִּי־סַרְתֶּם מַהֵר מֵאַחֲרֵי הַקּׂרֵא אֶתְכֶם בְּחֶסֶד הַמָּשִׁיחַ לִשְׁמׂעַ אֶל־בְּשׂוֹרָה זָרָה׃

ו

6

וְהִיא אֵינֶנָּה אַחֶרֶת רַק יֵשׁ אֲנָשִׁים הָעׂכְרִים אֶתְכֶם וַחֲפֵצִים לַהֲפׂךְ אֶת־בְּשׂוֹרַת הַמָּשִׁיחַ׃

ז

7

אֲבָל גַּם־אֲנַחְנוּ אוֹ־מַלְאָךְ מִן־הַשָּׁמַיִם אִם־יָבוֹא לְבַשֵׂר אֶתְכֶם בְּשׂוֹרָה מִבַּלְעֲדֵי זׂאת אֲשֶׁר בִּשַׂרְנוּ אֶתְכֶם חֵרֶם יִהְיֶה׃

ח

8

כַּאֲשֶׁר אָמַרְנוּ כְבָר כֵּן־אׂמַר עַתָּה עוֹד־הַפָּעַם אִישׁ כִּי־יְבַשֵׂר אֶתְכֶם בְּשׂוֹרָה מִבַּלְעֲדֵי אֲשֶׁר קִבַּלְתֶּם חֵרֶם יִהְיֶה׃

ט

9

כִּי הֲמִתְרַצֶּה אֶל־בְּנֵי אָדָם אָנׂכִי אוֹ אֶל־הָאֱלׂהִים אִם הַמְבַקֵּשׁ אָנׂכִי לִמְצׂא־חֵן בְּעֵינֵי בְנֵי־אָדָם הֵן בְּמָצְאִי חֵן בְּעֵינֵי בְנֵי־אָדָם לׂא־אֶהְיֶה עֶבֶד הַמָּשִׁיחַ׃

י

10

וּמוֹדִיעַ אֲנִי אֶתְכֶם אֶחָי כִּי הַבְּשׂוֹרָה אֲשֶׁר בִּשַׂרְתִּי לׂא־לְאָדָם הִיא׃

יא

11

כִּי אַף לׂא מֵאָדָם קִבַּלְתִּיהָ וְלׂא־מְלֻמֶּדֶת הִיא לִי כִּי אִם־בְּחֶזְיוֹן יֵשׁוּעַ הַמָּשִׁיחַ׃

יב

12

כִּי הֲלׂא־שְׁמַעְתֶּם אֶת־דַּרְכִּי מִלְפָנִים בַּדָּת הַיְּהוּדִית וְאֵת אֲשֶׁר־תַּכְלִית רְדִיפָה רָדַפְתִּי אֶת־עֲדַת אֱלׂהִים וְהֶחֱרַמְתִּיהָ׃

יג

13

וָאֱהִי הוֹלֵךְ וְחָזֵק בַּדָּת הַיְּהוּדִית עַל־רַבִּים מִבְּנֵי גִילִי בְּעַמִּי בְּקִנְאָתִי הַגְּדוֹלָה לְקַבָּלוֹת שֶׁל־אֲבוֹתָי׃

וּכְשֶׁהָיָה רָצוֹן מִלִּפְנֵי הָאֱלׂהִים הַמַּבְדִּיל אׂתִי מֵרֶחֶם אִמִּי וַיִּקְרָאֵנִי בְּחַסְדּוֹ׃

טו

15

לְגַלּוֹת בִּי אֶת־בְּנוֹ לְמַעַן אֲבַשְׂרֶנּוּ בַּגּוֹיִם אָז לׂא נוֹעַצְתִּי עִם־בָּשָׂר וָדָם׃

טז

16

גַּם לׂא־עָלִיתִי יְרוּשָׁלַיְמָה אֶל־אֲשֶׁר הָיוּ שְׁלִיחִים לְפָנָי כִּי אִם־הָלַכְתִּי לַעֲרָב וּמִשָּׁם שַׁבְתִּי אֶל־דַּמָּשֶׂק׃

יז

17

אַחֲרֵי־כֵן מִקֵּץ שָׁלשׁ שָׁנִים עָלִיתִי לִירוּשָׁלַיִם לִרְאוֹת אֶת־כֵּיפָא וָאֵשֵׁב עִמּוֹ חֲמִשָּׁה עָשָׂר יוֹם׃

יח

18

וְאַחֵר מִן־הַשְׁלִיחִים לׂא־רָאִיתִי זוּלָתִי אֶת־יַעֲקׂב אֲחִי אֲדׂנֵינוּ׃

יט

19

וַאֲשֶׁר אֲנִי כׂתֵב אֲלֵיכֶם הִנֵּה נֶגֶד הָאֱלׂהִים כִּי לׂא אֲכַזֵּב׃

כ

20

אַחֲרֵי־כֵן בָּאתִי אֶל־גְּלִילוֹת סוּרְיָא וְקִילִיקְיָא׃

כא

21

וּקְהִלּוֹת יְהוּדָה אֲשֶׁר בַּמָּשִׁיחַ הֵנָּה לׂא יָדְעוּ אֶת־פָּנָי׃

כב

22

רַק־זׂאת בִּלְבַד שָׁמְעוּ כִּי־הָרׂדֵף אׂתָנוּ מֵאָז עַתָּה מְבַשֵׂר אֶת־הָאֱמוּנָה אֲשֶׁר הֶחֱרִים מִלְּפָנִים׃

Chapter 2

אַחֲרֵי־כֵן מִקֵּץ אַרְבַּע עֶשְׂרֵה שָׁנָה שַׁבְתִּי וְעָלִיתִי לִירוּשָׁלַיִם עִם בַּר־נַבָּא וָאֶקַּח אִתִּי גַּם אֶת־טִיטוֹס׃

א

1

וָאַעַל שָׁמָּה עַל־פִּי מַחֲזֶה וְאֶת־הַבְּשׂוֹרָה אֲשֶׁר־קָרָאתִי בַגּוֹיִם שַׂמְתִּי לִפְנֵיהֶם וְשַׂמְתִּיהָ לִפְנֵי הַחֲשׁוּבִים שֶׁבָּהֶם פֶּן־תִּהְיֶה לָרִיק מְרוּצָתִי אֲשֶׁר אָרוּץ וַאֲשֶׁר־רָצְתִּי׃

ב

2

אֲבָל גַּם־טִיטוֹס אֲשֶׁר אִתִּי אַף כִּי־יְוָנִי הוּא לׂא הֻצְרַךְ לְהִמּוֹל׃

ג

3

מִפְּנֵי אֲחֵי הַשֶּׁקֶר הַמִּתְגַּנְּבִים בְּתוֹכֵנוּ אֲשֶׁר בָּאוּ לְרַגֵּל אֶת־חֵרוּתֵנוּ אֲשֶׁר־לָנוּ בְּיֵשׁוּעַ הַמָּשִׁיחַ לְמַעַן הַעֲבִידֵנוּ׃

ד

4

וַאֲנַחְנוּ לׂא־סַרְנוּ לְמִשְׁמַעְתָּם אַף־לׂא שָׁעָה אַחַת לְמַעַן אֲשֶׁר תַּעֲמׂד בְּקִרְבְּכֶם אֲמִתַּת הַבְּשׂוֹרָה׃

ה

5

וְהַנֶּחֱשָׁבִים לִהְיוֹת מָה הָיוּ מַה־שֶּׁהָיוּ אֵינִי חוֹשֵׁשׁ לָהֶם כִּי הָאֱלׂהִים לׂא יִשָׂא פְּנֵי־אִישׁ הֵן לִי לׂא־הוֹסִיפוּ הַחֲשׁוּבִים כָּל־דָּבָר׃

ו

6

וַתְּהִי לְהֶפֶךְ בִּרְאוֹתָם כִּי־הָפְקַדְתִּי אֲנִי עַל־בְּשׂוֹרַת הָעֲרֵלִים כַּאֲשֶׁר הָפְקַד כֵּיפָא עַל בְּשׂוֹרַת הַמּוּלִים׃

ז

7

כִּי־הַמֵּעִיר כֵּיפָא לְשָׁלְחוֹ אֶל־הַמּוּלִים הוּא־הֱעִירַנִי גַם־אׂתִי לְשָׁלְחֵנִי אֶל־הַגּוֹיִם׃

ח

8

וְכַאֲשֶׁר הִכִּירוּ יַעֲקׂב וְכֵיפָא וְיוֹחָנָן הַנֶּחֱשָׁבִים כְּעַמּוּדִים אֶת־הַחֶסֶד הַנִתָּן לִי נָתְנוּ לִי וּלְבַר־נַבָּא אֶת־יַד יְמִינָם וַנֵּאוֹת כִּי־נֵלֵךְ אֲנַחְנוּ אֶל־הַגּוֹיִם וְהֵמָּה אֶל־הַמּוּלִים׃

ט

9

רַק אִם־נִזְכּׂר אֶת־הָאֶבְיוֹנִים וְהוּא הַדָּבָר אֲשֶׁר שָׁקַדְתִּי לַעֲשׂוֹת׃

י

10

וְכַאֲשֶׁר בָּא כֵיפָא לְאַנְטְיוֹכְיָא הוֹכַחְתִּי דַרְכּוֹ אֶל־פָּנָיו כִּי נִמְצָא בוֹ עָוֶל׃

יא

11

כִּי לִפְנֵי בֹא אֲנָשִׁים מֵאֵת יַעֲקׂב אָכַל עִם־הַגּוֹיִם יַחְדָּו וּכְבֹאָם הָיָה מִתְרַחֵק וּפוֹרֵשׁ מֵהֶם מִפְּנֵי יִרְאָתוֹ אֶת־בְּנֵי הַמִּילָה׃

יב

12

וַיְכַחֲשׁוּ עִמּוֹ גַּם־שְׁאָר הַיְּהוּדִים עַד כִּי־נִדַּח גַּם־בַּר־נַבָּא אַחֲרֵי כַחֲשָׁם׃

יג

13

וּבִרְאוֹתִי שֶׁלּא יִשְּׁרוּ לֶכֶת כַּאֲמִתַּת הַבְּשׂוֹרָה אָמַרְתִּי אֶל־כֵּיפָא בְּאָזְנֵי כֻלָּם אִם־אַתָּה הַיְּהוּדִי כְּנָכְרִי תִתְנַהֵג וְלׂא כִיהוּדִי מַדּוּעַ תַּכְרִיחַ אֶת־הַגּוֹיִם לְהִתְנַהֵג כַּיְּהוּדִים׃

יד

14

הֵן מִזֶּרַע הַיְּהוּדִים אֲנַחְנוּ וְלׂא חַטָּאִים מִן־הַגּוֹיִם׃

טו

15

אֲבָל מִפְּנֵי שֶׁיּוֹדְעִים אֲנַחְנוּ שֶׁלּא־יִצְדַּק אָדָם מִתּוֹךְ מַעֲשֵׂי הַתּוֹרָה כִּי אִם־בֶּאֱמוּנַת יֵשׁוּעַ הַמָּשִׁיחַ כָּל־בָּשָׂר׃

טז

16

וְאִם נִמָּצֵא גַם־אֲנַחְנוּ חַטָּאִים בְּבַקְשֵׁנוּ לְהִצָּדֵק בַּמָּשִׁיחַ הִנֵּה הַמָּשִׁיחַ מְשָׁרֵת הַחֵטְא חָלִילָה׃

יז

17

כִּי אִם־אָשׁוּב וְאֶבְנֶה אֶת־אֲשֶׁר סָתַרְתִּי אֶעֱשֶׂה אֶת־עַצְמִי לְפשֵׁעַ׃

יח

18

כִּי־מַתִּי אֲנִי לַתּוֹרָה עַל־יְדֵי הַתּוֹרָה לְמַעַן אֶחְיֶה לֵאלׂהִים׃

יט

19

עִם־הַמָּשִׁיחַ נִצְלַבְתִּי וְאָנׂכִי לׂא אֶחְיֶה עוֹד כִּי אִם־הַמָּשִׁיחַ הוּא חַי בְּקִרְבִּי וַאֲשֶׁר אֲנִי חַי עַתָּה בַּבָּשָׂר חַי אֲנִי בֶּאֱמוּנַת בֶּן־אֱלׂהִים אֲשֶׁר אֲהָבַנִי וַיִּתֵּן אֶת־נַפְשׁוֹ בַּעֲדִי׃

כ

20

לׂא אֲבַטֵּל אֶת־חֶסֶד הָאֱלׂהִים כִּי אִלּוּ יֵשׁ צְדָקָה עַל־יְדֵי הַתּוֹרָה אַךְ לַשָּׁוְא מֵת הַמָּשִׁיחַ׃

Chapter 3

אֲהָהּ גָּלָטִים חַסְרֵי דָעַת מִי הִתְעָה אֶתְכֶם בִּכְשָׁפָיו (לְבִלְתִּי שְׁמׂעַ אֶל־הָאֱמֶת) אַחֲרֵי אֲשֶׁר צֻיַּר יֵשׁוּעַ הַמָּשִׁיחַ הַצָּלוּב לְנֶגֶד עֵינֵיכֶם׃

א

1

רַק זׂאת אֶחְפּׂץ לִלְמׂד מִכֶּם הַאִם מִמַּעֲשֵׂי הַתּוֹרָה קִבַּלְתֶּם אֶת־הָרוּחַ אוֹ מִשְּׁמוּעַת הָאֱמוּנָה׃

ב

2

הַסְכָלִים אַתֶּם כָּל־כָּךְ הַחִלּוֹתֶם בָּרוּחַ וְעַתָּה תְּכַלּוּ בַבָּשָׂר׃

ג

3

הֲכָזׂאת נַעֲנֵיתֶם לָרִיק אִם־אָמְנָם לָרִיק׃
