Jacob (aka James)

Chapter 1

יַעֲקׂב עֶבֶד אֱלׂהִים וַאֲדׂנֵינוּ יֵשׁוּעַ הַמָּשִׂיחַ שׁׂאֵל לִשְׁלוֹם שְׁנֵים עָשָׂר הַשְּׁבָטִים שֶּׁבַּגּוֹלָה׃

א

1

אַךְ לְשִׂמְחָה חִשְׁבוּ לָכֶם אֶחָי כַּאֲשֶׁר תָּבֹאוּ בְּנִסְיׂנוֹת שׁוֹנִים׃

ב

2

מִפְּנֵי שֶׁיּׂדְעִים אַתֶּם כִּי־בֹחַן אֱמוּנַתְכֶם מֵבִיא לִידֵי סַבְלָנוּת׃

ג

3

וְהַסַּבְלָנוּת שְׁלֵמָה תִּהְיֶה בְּפָעֳלָהּ לְמַעַן תִּהְיוּ שְׁלֵמִים וּתְמִימִים וְלׂא תַחְסְרוּ כָּל־דָּבָר׃

ד

4

וְאִישׁ מִכֶּם כִּי יֶחְסַר חָכְמָה יְבַקְשֶׁנָּה מֵאֱלׂהִים הַנּוֹתֵן לַכּׂל בִּנְדִיבָה וּבְלׂא הוֹנָאַת דְּבָרִים וְתִנָּתֵן לוֹ׃

ה

5

וּבִלְבַד שֶׁיְּבַקֵּשׁ בֶּאֱמוּנָה וּבִבְלִי סָפֵק כִּי־בַעַל סָפֵק דּוֹמֶה לְגַל הַיָּם נִשָּׂא וּמְטׂרָף בָּרוּחַ׃

ו

6

וְהָאִישׁ הַהוּא אַל־יְדַמֶּה בְנַפְשׁוֹ כִּי־יִּשָּׂא דָבָר מֵאֵת יְהוָֹה׃

ז

7

אִישׁ אֲשֶׁר חָלַק לִבּוֹ הֲפַכְפָּךְ הוּא בְּכָל־דְּרָכָיו׃

ח

8

אֲבָל הָאָח הַשָּׁפֵל יִתְהַלֵּל בְּרוֹמֲמֻתוֹ׃

ט

9

וְהֶעָשִׁיר יִתְהַלֵּל בְּשִׁפְלוּתוֹ כִּי יַעֲבֹר כְּצִיץ הֶחָצִיר׃

י

10

כִּי זָרַח הַשֶּׁמֶשׁ בְּחַמָּתוֹ וַיְיַבֵּשׁ אֶת־הֶחָצִיר וַיִּבֹּל צִיצוֹ וְחֵן מַרְאֵהוּ אָבָד כֵּן יִבּוֹל הֶעָשִׁיר בַּהֲלִיכוֹתָיו׃

יא

11

אַשְׁרֵי הָאִישׁ הָעׂמֵד בְּנִסְיוֹנוֹ כִּי כַּאֲשֶׁר נִבְחַן יִשָּׂא עֲטֶרֶת הַחַיִּים אֲשֶׁר־הִבְטִיחַ יְהוָֹה לְאׂהֲבָיו׃

יב

12

אַל־יׂאמַר הַמְנֻסֶּה הָאֱלׂהִים נִסָּנִי כִּי הָאֱלׂהִים אֵינֶנּוּ מְנֻסֶּה בָּרָע וְהוּא לׂא־יְנַסֶּה אִישׁ׃

יג

13

כִּי אִם־יְנֻסֶּה כָל־אִישׁ בְּתַאֲוַת נַפְשׁוֹ אֲשֶׁר תְּסִיתֵהוּ וּתְפַתֵּהוּ׃

יד

14

וְאַחֲרֵי־כֵן הָרְתָה הַתַּאֲוָה וַתֵּלֶד חֵטְא וְהַחֵטְא כִּי נִגְמַר יוֹלִיד אֶת־הַמָּוֶת׃

טו

15

אַל־תִּתְעוּ אַחַי אֲהוּבָי׃

טז

16

כָּל־מַתָּת טוֹבָה וְכָל־מַתָּנָה שְׁלֵמָה תֵּרֵד מִמַּעַל מֵאֵת אֲבִי הָאוֹרוֹת אֲשֶׁר חִלּוּף וְכָל־צֵל שִׁנּוּי אֵין־עִמּוֹ׃

יז

17

הוּא בְחֶפְצוֹ יָלַד אוֹתָנוּ בִּדְבַר הָאֱמֶת לִהְיוֹת כְּמוֹ רֵאשִׁית בִּכּוּרֵי יְצוּרָיו׃

יח

18

עַל־כֵּן אַחַי אֲהוּבַי יְהִי כָל־אִישׁ מָהִיר לִשְׁמׂעַ קָשֶׁה לְדַבֵּר וְקָשֶׁה לִכְעוֹס׃

יט

19

כִּי־כַעַס אָדָם לׂא יִפְעַל צִדְקַת אֱלׂהִים׃

כ

20

לָכֵן הָסִירוּ מֵעֲלֵיכֶם כָּל־טִנּוּף וְתַרְבּוּת רָעָה וְקַבְּלוּ בַעֲנָוָה אֶת־הַדָּבָר הַנָּטוּעַ בָּכֶם אֲשֶׁר יָכׂל לְהוֹשִׁיעַ אֶת־נַפְשׁׂתֵיכֶם׃

כא

21

וִהְיוּ עׂשֵׂי הַדָּבָר וְלׂא שׁׂמְעָיו בִּלְבָד לְרַמּוֹת אֶת־נַפְשְׁכֶם׃

כב

22

כִּי הָאִישׁ הַשּׁׂמֵעַ אֶת־הַדָּבָר וְאֵין עׂשֵׂהוּ דּׂמֶה לְאִישׁ מַבִּיט אֶת־תּׂאַר הֲוָיָתוֹ בְּמַרְאָה׃

כג

23

כִּי הָאִישׁ הַשּׁׂמֵעַ אֶת־הַדָּבָר וְאֵין עׂשֵׂהוּ דּׂמֶה לְאִישׁ מַבִּיט אֶת־תּׂאַר הֲוָיָתוֹ בְּמַרְאָה׃

כג

23

כִּי הִבִּיט אֶל־מַרְאֵהוּ וַיֵּלֶךְ לוֹ וּבְרֶגַע שָׁכַח מַה־תָּאֳרוֹ׃

כד

24

אֲבָל הַמַּשְׁקִיף בַּתּוֹרָה הַשְּׁלֵמָה תּוֹרַת הַחֵרוּת וּמַחֲזִיק בָּהּ אֲשֶׁר אֵינֶנּוּ שׁׂמֵעַ וְשָׁכֵחַ כִּי אִם־עׂשֶׂה בְּפׂעַל אַשְׁרֵי הָאִישׁ הַהוּא בְּמַעֲשֵׂהוּ׃

כה

25

אִישׁ מִכֶּם אִם־יְדַמֶּה לִהְיוֹת עׂבֵד אֱלׂהִים וְאֵינֶנּוּ שָׂם רֶסֶן לִלְשׁׂנוֹ כִּי אִם־מַתְעֶה הוּא אֶת־לְבָבוֹ עֲבֹדָתוֹ אַךְ־לָרִיק תִּהְיֶה׃

כו

26

זׂאת הִיא הָעֲבוֹדָה הַטְּהוֹרָה וְהַבָּרָה לִפְנֵי הָאֱלׂהִים אָבִינוּ לִפְקׂד אֶת־הַיְתוֹמִים וְהָאַלְמָנוֹת בְּצָרָתָם וְלִשְׁמׂר נַפְשׁוֹ בְּנִקָּיוֹן מֵחֶלְאַת הָעוֹלָם׃

כז

27

Chapter 2

אַחַי אַל־יְהִי מַשּׂא פָנִים בֶּאֱמוּנַתְכֶם בְּיֵשׁוּעַ הַמָּשִׁיחַ אֲדׂנֵינוּ אֲדוֹן הַכָּבוֹד׃

א

1

כִּי אִם־יָבוֹא אִישׁ לְבֵית הַכְּנֵסֶת שֶׁלָּכֶם וְטַבְּעׂת זָהָב עַל־יָדָיו וְהוּא לָבוּשׁ לְבֻשֵׁי מִכְלוֹל וּבָא שָׁמָּה גַּם־אִישׁ עָנִי בִּבְגָדִים צוֹאִים׃

ב

2

וּפְנִיתֶם אֶל־הַלָּבוּשׁ לְבֻשֵׁי מִכְלוֹל וַאֲמַרְתֶּם לוֹ שֶׁב־לְךָ הֵנָּה בְּכָבוֹד וְלֶעָנִי תׂאמְרוּ עֲמָד־שָׁם אוֹ שֶׁב־פּׂה מִתַּחַת לַהֲדׂם רַגְלָי׃

ג

3

הֲלׂא לֵב וָלֵב לָכֶם וְהִנְּכֶם שׁׂפְטִים בַּעֲלֵי מַחֲשָׁבוֹת רָעוֹת׃

ד

4

שִׁמְעוּ אַחַי אֲהוּבָי הֲלׂא בַעֲנִיֵּי הָעוֹלָם הַזֶּה בָּחַר הָאֱלׂהִים לִהְיוֹת עֲשִׁירִים בָּאֱמוּנָה וְיׂרְשֵׁי הַמַּלְכוּת אֲשֶׁר הִבְטִיחַ לְאׂהֲבָיו׃

ה

5

וְאַתֶּם הִכְלַמְתֶּם אֶת־הֶעָנִי הֲלׂא הָעֲשִׁירִים הֵם הָעׂשְׁקִים אֶתְכֶם וְהֵם הַסּׂחֲבִים אֶתְכֶם אֶל־בָּתֵּי דִין׃

ו

6

הֲלׂא הֵם הַמְגַדְּפִים אֶת־הַשֵּׁם הַטּוֹב הַנִּקְרָא עֲלֵיכֶם׃

ז

7

הֵן בַּעֲשׂוֹתְכֶם אֶת־הַמִּצְוָה הַמּׂלֶכֶת עַל־כֻּלָּן כְּפִי הַכָּתוּב וְאָהַבְתָּ לְרֵעֲךָ כָּמוֹךָ תֵּיטִיבוּ לַעֲשׂוֹת׃

ח

8

אוּלָם אִם־תִּשְׂאוּ פָנִים חַטָּאִים אַתֶּם וְהַתּוֹרָה תוֹכִיחֲכֶם כְּעׂבְרִים עָלֶיהָ׃

ט

9

כִּי אִישׁ אֲשֶׁר יְקַיֵּם אֶת־כָּל־הַתּוֹרָה וְנִכְשַׁל בְּאַחַת מִמִּצְוֹתֶיהָ נִדּוֹן עַל־כֻּלָּן׃

י

10

כִּי הָאׂמֵר לׂא תִנְאָף הוּא הָאׂמֵר לׂא תִרְצָח וְאִם־אֵינְךָ נׂאֵף וְאַתָּה רוֹצֵחַ הִנְּךָ עׂבֵר עַל־הַתּוֹרָה׃

יא

11

כֵּן דַּבְּרוּ וְכֵן עֲשׂוֹוֹ כְּמִי שֶׁעֲתִידִים לְהִשָּׁפֵט עַל־פִּי תּוֹרַת הַחֵרוּת׃

יב

12

כִּי אֵין חֶסֶד בַּמִּשְׁפָּט לַאֲשֶׁר לׂא־עָשָׂה חָסֶד וְהַחֶסֶד יִתְגָּאֶה עַל־הַמִּשְׁפָּט׃

יג

13

אַחַי מַה־יּוֹעִיל לְאִישׁ שֶׁיּׂאמַר כִּי אֱמוּנָה בוֹ וּמַעֲשִׂים אֵין בּוֹ הֲתוּכַל הָאֱמוּנָה לְהוֹשִׁיעוֹ׃

יד

14

אָח אוֹ אָחוֹת כִּי־יִהְיוּ בְעֵירׂם וְאֵין לָהֶם לֶחֶם חֻקָּם׃

טו

15

וְאִישׁ מִכֶּם יׂאמַר אֲלֵיהֶם לְכוּ לְשָׁלוֹם וְהִתְחַמֲמוּ וּשְׂבָעוּ וְלׂא־תִתְּנוּ לָהֶם צָרְכֵי גוּפָם מַה־תּוֹעִיל זׂאת׃

טז

16

כָּכָה גַּם־הָאֱמוּנָה אִם־אֵין בָּהּ מַעֲשִׂים מֵתָה הִיא בְּעַצְמָהּ׃

יז

17

וְאִם־יׂאמַר אִישׁ אַתָּה אֱמוּנָה בְךָ וְלִי מַעֲשִׂים הַרְאֵנִי נָא אֶת־אֱמוּנָתְךָ בִּבְלִי מַעֲשִׂים וְאַרְאֲךָ אֲנִי מִתּוֹךְ מַעֲשַׂי אֶת־אֱמוּנָתִי׃

יח

18

אַתָּה מַאֲמִין שֶׁהָאֱלׂהִים אֶחָד הוּא הֱטִיבוֹתָ לְהַאֲמִין גַּם הַשֵּׁדִים מַאֲמִינִים בּוֹ וְרׂעֲדִים׃

יט

19

וְאַתָּה אִישׁ־בָּעַר הֲתַחְפּׂץ לָדַעַת כִּי הָאֱמוּנָה בְּאֵין מַעֲשִׂים מֵתָה הִיא׃

כ

20

אַבְרָהָם אָבִינוּ הֲלׂא בְּמַעֲשָׂיו נִצְדָּק בְּהַעֲלׂתוֹ אֶת־יִצְחָק בְּנוֹ עַל־הַמִּזְבֵּחַ׃

כא

21

הִנְּךָ רׂאֶה כִּי־הָאֱמוּנָה עׂזֶרֶת לְמַעֲשָׂיו וּמִתּוֹךְ הַמַּעֲשִׂים הָשְׁלְמָה הָאֱמוּנָה׃

כב

22

וַיִּמָּלֵא הַכָּתוּב הָאׂמֵר וְהֶאֱמִן אַבְרָהָם בַּיהוָֹה וַתֵּחָשֶׁב־לוֹ לִצְדָקָה וַיִּקָּרֵא אׂהֵב יְהוָֹה׃

כג

23

הִנְּכֶם רׂאִים כִּי בְּמַעֲשִׂים יִצְדַּק הָאִישׁ וְלׂא בֶאֱמוּנָה לְבַדָּהּ׃

כד

24

וְכֵן גַּם־רָחָב הַזּוֹנָה הֲלׂא נִצְדְּקָה בְּמַעֲשִׂים בְּאָסְפָהּ אֶת־הַמַּלְאָכִים אֶל־בֵּיתָהּ וַתְּשַׁלְּחֵם בְּדֶרֶךְ אַחֵר׃

כה

25

כִּי כַּאֲשֶׁר הַגּוּף מִבְּלִי נְשָׁמָה מֵת הוּא כֵּן גַּם־הָאֱמוּנָה מִבְּלִי־מַעֲשִׂים מֵתָה׃

כו

Chapter 3
