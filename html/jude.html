Jude

Chapter 1

יְהוּדָה עֶבֶד יֵשׁוּעַ הַמָּשִׁיחַ וַאֲחִי יַעֲקׂב אֶל־הַמְקׂרָאִים אֲשֶׁר הֵם אֲהוּבִים בֵּאלׂהִים הָאָב וּשְׁמוּרִים לְיֵשׁוּעַ הַמָּשִׁיחַ׃

א

1

רַחֲמִים וְשָׁלוֹם וְאַהֲבָה יִהְיוּ לָכֶם לְמַכְבִּיר׃

ב

2

חֲבִיבַי בְּשָׁקְדִי לִכְתּׂב אֲלֵיכֶם עַל־דְּבַר הַתְּשׁוּעָה הָאַחַת לְכֻלָּנוּ הֱצִיקַתְנִי רוּחִי לְעוֹרֵר אֶתְכֶם בְּמִכְתָּב אֲשֶׁר תִלָּחֲמוּ לָאֱמוּנָה הַמְּסוּרָה פַּעַם אַחַת לַקְּדוֹשִׁים׃

ג

3

כִּי הִתְגַּנְּבוּ לָבוֹא אֲנָשִׁים אֲשֶׁר נִגְזַר דִּינָם מֵאָז אַנְשֵׁי רֶשַׁע הַהׂפְכִים אֶת־חֶסֶד אֱלׂהֵינוּ לְזִמָּה וְכׂפְרִים בֵּאלׂהִים הַמּשֵׁל הַיָּחִיד וּבַאֲדׂנֵינוּ יֵשׁוּעַ הַמָּשִׁיחַ׃

ד

4

וְיֵשׁ אֶת־נַפְשִׁי לְהַזְכִּיר אֶתְכֶם אֶת־אֲשֶׁר כְּבָר יְדַעְתֶּם כִּי הוֹשִׁיעַ יְהוָֹה אֶת־הָעָם מִמִּצְרַיִם וַיַּשְׁמֵד אַחַר כֵּן אֶת־אֲשֶׁר לׂא הֶאֱמִינוּ׃

ה

5

וְהַמַּלְאָכִים אֲשֶׁר לׂא־שָׁמְרוּ אֶת־מִשְׂרָתָם כִּי אִם־עָזְבוּ אֶת־מְעוֹנָם שְׁמָרָם בְּמוֹסְרוֹת עוֹלָם בַּאֲפֵלָה לְמִשְׁפַּט הַיּוֹם הַגָּדוֹל׃

ו

6

כַּאֲשֶׁר נִהְיוּ לְמָשָׁל סְדוֹם וַעֲמׂרָה וּבְנׂתֵיהֶן אֲשֶׁר הִזְנוּ כְּמוֹהֶם וַתֵּלַכְנָה אַחֲרֵי בָּשָׂר זָר כִּי־נִמְסְרוּ לְמוּסַר אֵשׁ עוֹלָם׃

ז

7

וְכֵן בַּעֲלֵי הַחֲלׂמוֹת הָאֵלֶּה מְטַמְּאִים אֶת־הַבָּשָׂר וְאֶת־הַמֶּמְשָׁלָה יְנָאֲצוּ וְאֶת־הַשְּׂרָרוֹת יְחָרֵפוּ׃

ח

8

וּמִיכָאֵל שַׂר הַמַּלְאָכִים בְּהִתְוַכְּחוֹ עִם־הַשָּׂטָן וַיָּרֶב אִתּוֹ עַל־אׂדוֹת גְּוִיַּת מׂשֶׁה לׂא־מְלָאוֹ לִבּוֹ לַחֲרׂץ מִשְׁפַּט גִּדּוּפִים כִּי אִם־אָמַר יִגְעַר יְהוָֹה בָּךְ׃

ט

9

וְאֵלֶּה מְחָרְפִים אֶת־אֲשֶׁר לׂא־יָדָעוּ וּבַדְּבָרִים אֲשֶׁר יָבִינוּ מֵחָק־טִבְעָם כַּבְּהֵמוֹת הַסְּכָלוֹת בָּהֵמָּה יַשְׁחִיתוּ אֶת־נַפְשָׁם׃

י

10

אוֹי לָהֶם כִּי־בְדֶרֶךְ קַיִן הָלָכוּ וַיִּדַּחוּ בְּמַדּוּחֵי בִלְעָם עֵקֶב שָׂכָר וּבִמְרִי קׂרַח אָבָדוּ׃

יא

11

הֲלׂא צוּרֵי מִכְשׁוֹל הֵמָּה בִּסְעוּדׂת אַהֲבָה שֶׁלָּכֶם אׂכְלִים וְשׁׂתִים עִמָּכֶם בְּקַלּוּת רׂאשׁ וְרׂעִים אֶת־נַפְשָׁם עָבִים הֵם בְּלִי מַיִם נִדָּפִים מִפְּנֵי רוּחַ עֲצֵי־חׂרֶף בְּאֵין פְּרִי אֲשֶׁר מֵתוּ פַעֲמַיִם וְנֶעֱקָרוּ׃

יב

12

מִשְׁבְּרֵי־יָם עַזִּים אֲשֶׁר יִגְרְשׁוּ בָשְׁתָּם כּׂכָבִים תּׂעִים אֲשֶׁר־חשֶׁךְ אֲפֵלָה צָפוּן לָהֶם עֲדֵי־עַד׃

יג

13

וְגַם־חֲנוֹךְ הַשְּׁבִיעִי לְאָדָם נִבָּא עֲלֵיהֶם לֵאמׂר הִנֵּה יְהוָֹה בָּא בְּרִבֲבֹת קְדׂשָׁיו׃

יד

14

לַעֲשׂוֹת שְׁפָטִים בְּכֻלָּם וּלְהוֹכִיחַ כָּל־רִשְׁעֵי־אֶרֶץ עַל־כָּל־מַעֲשֵׂי רִשְׁעָתָם וְעַל־כָּל־הַקָּשׁוֹת אֲשֶׁר דִּבְּרוּ עָלָיו חַטָּאִים אַנְשֵׁי רֶשַׁע׃

טו

15

אֵלֶּה הֵם הָרׂגְנִים וְהַמִּתְאוֹנֲנִים הַהׂלְכִים אַחֲרֵי תַאֲוֹתֵיהֶם אֲשֶׁר פִּיהֶם יְדַבֵּר עָתָק הׂדְרֵי פָנִים עֵקֶב שׁׂחַד׃

טז

16

וְאַתֶּם חֲבִיבַי זִכְרוּ אֶת־הַדְּבָרִים הַנֶּאֱמָרִים מִקֶּדֶם בְּיַד שְׁלִיחֵי אֲדׂנֵינוּ יֵשׁוּעַ הַמָּשִׁיחַ׃

יז

17

בְּדַבְּרָם אֲלֵיכֶם לֵאמׂר הִנֵּה בְּאַחֲרִית הַיָּמִים יָבֹאוּ לֵצִים הַהׂלְכִים אַחֲרֵי תַּאֲוֹת רִשְׁעָם׃

יח

18

אֵלֶּה הֵם הַפּׂרְשִׁים מִן־הַצִּבּוּר בַּעֲלֵי נֶפֶשׁ וְרוּחַ אֵין בָּהֶם׃

יט

19

וְאַתֶּם חֲבִיבַי הִבָּנוּ בֶּאֱמוּנַתְכֶם הַנַּעֲלָה עַל־כּׂל בִּקְדֻשָּׁתָהּ וְהִתְפַּלֲלוּ בְּרוּחַ הַקּׂדֶשׁ׃

כ

20

וּשְׁמַרְתֶּם אֶת־נַפְשׁׂתֵיכֶם בְּאַהֲבַת אֱלׂהִים וְחִכִּיתֶם לְרַחֲמֵי אֲדׂנֵינוּ יֵשׁוּעַ הַמָּשִׁיחַ לְחַיֵּי עוֹלָם׃

כא

21

הַבְדִּילוּ אֶת־אֵלֶה וְהִתְנַהֲגוּ עִמָּהֶם בְּרַחֲמִים׃

כב

22

וְאֶת־אֵלֶּה תּוֹשִׁיעוּ בְאֵימָה וְחִלַּצְתֶּם אׂתָם מִתּוֹךְ הָאֵשׁ וּמְאַסְתֶּם גַּם־אֶת־הַלְּבוּשׁ הַמְגׂאָל בְּחֶלְאַת הַבָּשָׂר׃

כג

23

וַאֲשֶׁר יָכוֹל לִשְׁמָרְכֶם מִמִּכְשׁוֹל וּלְהַעֲמִיד אֶתְכֶם בְּשָׂשׂוֹן תְּמִימִים לִפְנֵי כְבוֹדוֹ׃

כד

24

הָאֱלׂהִים (הֶחָכָם) הַיָּחִיד הַמּוֹשִׁיע אׂתָנוּ בְּיֵשׁוּעַ הַמָּשִׁיחַ אֲדׂנֵינוּ לוֹ הַכָּבוֹד וְהַגְּדֻלָּה וְהָעׂז וְהַמֶּמְשָׁלָה לִפְנֵי כָל־עוֹלָם גַּם עַתָּה גַּם לְעוֹלְמֵי עַד אָמֵן׃

כה

25
