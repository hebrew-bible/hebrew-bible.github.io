Philippians

Chapter 1

פּוֹלוֹס וְטִימוֹתִיּוֹס עַבְדֵי הַמָּשִׁיחַ יֵשׁוּעַ אֶל־כָּל־הַקְּדשִׁים בַּמָּשִׁיחַ יֵשׁוּעַ אֲשֶׁר הֵם בְּפִילִפִּי עִם־הַהֶגְמוֹנִם וְהַשַּׁמָּשִׁים׃

א

1

חֶסֶד לָכֶם וְשָׁלוֹם מֵאֵת הָאֱלׂהִים אָבִינוּ וַאֲדׂנֵינוּ יֵשׁוּעַ הַמָּשִׁיחַ׃

ב

2

אוֹדֶה לֵאלׂהָי מִדֵּי זָכְרִי אֶתְכֶם׃

ג

3

וְתָמִיד בְּכָל־תְּפִלּוֹתַי אֶתְחַנֵּן בְּשִׂמְחָה בְּעַד כֻּלְּכֶם׃

ד

4

עַל־הִתְחַבְּרוּתְכֶם אֶל־הַבְּשׂוֹרָה לְמִן־הַיּוֹם הָרִאשׁוֹן וְעַד־הֵנָּה׃

ה

5

וּבָטוּחַ אֲנִי בַּמַּתְחִיל בָּכֶם הַמַּעֲשֶׂה הַטּוֹב שֶׁגַּם־יִגְמְרֶנּוּ עַד־יוֹם יֵשׁוּעַ הַמָּשִׁיחַ׃

ו

6

כָּרָאוּי לִי לַחֲשׁׂב לְכֻלְּכֶם בַּעֲבוּר שְׂאֵתִי אֶתְכֶם בִּלְבָבִי בְּמוֹסֵרַי וּבְהַצְדִּיקִי וְחַזְּקִי אֶת־הַבְּשׂוֹרָה כִּי אַתֶּם כֻּלְּכֶם חֲבֵרַי בֶּחָסֶד׃

ז

7

כִּי הָאֱלׂהִים לְעֵד לִי אֲשֶׁר נִכְסַפְתִּי לְכֻלְּכֶם בְּאַהֲבַת יֵשׁוּעַ הַמָּשִׁיחַ׃

ח

8

וְעַל־זׂאת מִתְפַּלֵּל אָנׂכִי שֶׁעוֹד תִּרְבֶּה וְתִגְדַּל אַהֲבַתְכֶם לְהַשְׂכֵּל וּלְכָל־דָּעַת׃

ט

9

לְמַעַן תֵּדְעוּ בֵּין טוֹב לָרָע וִהְיִיתֶם זַכִּים וּבְלִי מִכְשׁוֹל עַד־יוֹם הַמָּשִׁיחַ׃

י

10

מְלֵאִים פְּרִי הַצְּדָקָה עַל־יְדֵי יֵשׁוּעַ הַמָּשִׁיחַ לִכְבוֹד אֱלׂהִים וְלִתְהִלָּתוֹ׃

יא

11

וְהִנְנִי מוֹדִיעַ אֶתְכֶם אֶחָי כִּי מַה־שֶּׁבָּא עָלַי לׂא בָא אֶלָּא לְיִתְרוֹן הַבְּשׂוֹרָה׃

יב

12

יג

13

וְרׂב הָאַחִים בַּאֲדׂנֵינוּ הוֹסִיפוּ אׂמֶץ בְּמוֹסְרוֹתַי וְהִתְחַזְּקוּ בִלְבָבָם לְדַבֵּר אֶת־הַדָּבָר בְּלִי־פָחַד׃

יד

14

הֵן יֵשׁ מַכְרִיזִים אֶת־הַמָּשִׁיחַ מִקִּנְאָה וָרִיב וְיֵשׁ מַכְרִיזִים מִמַּחֲשָׁבָה טוֹבָה׃

טו

15

אֵלֶּה מִמְּרִיבָה מַכְרִיזִים אֶת־הַמָּשִׁיחַ לׂא בְלֵב־טָהוֹר כִּי חשְׁבִים לְהוֹסִיף צָרָה עַל־מוֹסֵרָי׃

טז

16

וְאֵלֶּה מֵאַהֲבָה בַּאֲשֶׁר הֵם יׂדְעִים כִּי־נָתוּן אָנׂכִי לְהַצְדִּיק אֶת־הַבְּשׂוֹרָה׃

יז

17

אֶלָּא מַה־בְּכָךְ וּבִלְבַד שֶׁיֻּכְרַז הַמָּשִׁיחַ בֵּין בַּעֲלִילָה וּבֵין בֶּאֱמֶת בָּזׂאת אֲנִי שָׂמֵחַ וְגַם אֱהְיֶה שָׂמֵח׃

יח

18

כִּי יוֹדֵעַ אֲנִי שֶׁגַּם זׂאת תִּהְיֶה־לִּי לִישׁוּעָה בִּתְפִלַּתְכֶם וּבְעֶזְרַת רוּחַ יֵשׁוּעַ הַמָּשִׁיחַ׃

יט

19

וְאוֹחִיל וַאֲקַוֶּה שֶׁלּא אֵבוֹשׁ מִכָּל־דָּבָר כִּי אִם־בְּכָל־בִּטָּחוֹן כַּאֲשֶׁר מֵאָז כֵּן עַתָּה יִתְגַּדֵּל הַמָּשִׁיחַ בְּגוּפִי אִם־בְּחַיַּי אִם־בְּמוֹתִי׃

כ

20

כִּי הַמָּשִׁיחַ הוּא חַיָּי וְהַמָּוֶת רֶוַח־לִי׃

כא

21

אֶלָּא אִם־חַיַּי בַּבָּשָׂר יוֹסִיף־לִי פְּרִי עֲמָלִי לׂא אֵדַע בְּמָה אֶבְחָר׃

כב

22

כִּי אָחוּז אֲנִי מִשְּׁנֵי צְדָדִים נַפְשִׁי אִוְּתָה לְהִפָּטֵר וְלִהְיוֹת עִם־הַמָּשִׁיחַ כִּי־זֶה הַמֻּבְחָר׃

כג

23

אֲבָל עֲמׂד עוֹד בַּבָּשָׂר צָרִיךְ יׂתֵר בַּעֲבוּרְכֶם׃

כד

24

וַאֲנִי בֹטֵחַ וְיׂדֵעַ שֶׁאִוָּתֵר וְאֶעֱמׂד עִם־כֻּלְּכֶם יָחַד לְאַמֵּץ אֶתְכֶם וּלְהַגְדִּיל שִׂמְחַת אֱמוּנַתְכֶם׃

כה

25

לְמַעַן תִּרְבֶּה עַל־יָדִי תְּהִלַּתְכֶם בַּמָּשִׁיחַ יֵשׁוּעַ בְּשׁוּבִי לָבוֹא אֲלֵיכֶם׃

כו

26

רַק הִתְנַהֲגוּ כָּרָאוּי לִבְשׂוֹרַת הַמָּשִׁיחַ לְמַעַן אֶשְׁמַע עֲלֵיכֶם אִם־בְּבוֹאִי לִרְאוֹתְכֶם אִם־בִּהְיוֹתִי רָחוֹק שֶׁקַּיָּמִים אַתֶּם בְּרוּחַ אַחַת וְעׂזְרִים אׂתִי נֶפֶשׁ אַחַת לְהִלָּחֵם בְּעַד אֱמוּנַת הַבְּשׂוֹרָה׃

כז

27

וְאֵינְכֶם חֲרֵדִים מְאוּמָה מִפְּנֵי הַמִּתְקוֹמֲמִים אֲשֶׁר זׂאת לָהֶם אוֹת לְאָבְדָם וְלָכֶם לִישׁוּעַתְכֶם וְגַם־זׂאת מֵאֵת הָאֱלׂהִים׃

כח

28

כִּי־לָכֶם נִתַּן לְהִתְעַנּוֹת בְּעַד הַמָּשִׁיחַ וְלׂא־לְהַאֲמִין בּוֹ בִּלְבָד׃

כט

29

כִּי גַם־לָכֶם הַמִּלְחָמָה אֲשֶׁר רְאִיתֶם בִּי וַאֲשֶׁר עַתָּה שׁׂמְעִים אַתֶּם עָלָי׃

Chapter 2

לָכֵן אִם־יֵשׁ תּוֹכַחַת בַּמָּשִׁיחַ אִם־תַּנְחוּמוֹת הָאַהֲבָה אִם־חֶבְרַת רוּחַ אִם־רַחֲמִים וְחֶמְלָה׃

א

1

הַשְׁלִימוּ־נָא אֶת־שִׂמְחָתִי בִּהְיוֹת לָכֶם לֵב אֶחָד וְאַהֲבָה אַחַת וְנֶפֶשׁ אַחַת וְרָצוֹן אֶחָד׃

ב

2

וְאַל־תַּעֲשׂוּ דָבָר בְּדֶרֶךְ מְרִיבָה אוֹ כְּבוֹד שָׁוְא כִּי אִם־בְּשִׁפְלוּת רוּחַ יַחְשׁׂב אִישׁ אֶת־רֵעֵהוּ גָּדוֹל מִמֶּנּוּ׃

ג

3

כָּל־אֶחָד אַל־יִדְאַג לַאֲשֶׁר־לוֹ לְבַדּוֹ כִּי אִם־גַּם לַאֲשֶׁר לַחֲבֵרוֹ׃

ד

4

כִּי יְהִי לְבַבְכֶם כִּלְבַב הַמָּשִׁיחַ יֵשׁוּעַ׃

ה

5

אֲשֶׁר אַף כִּי־הָיָה בִּדְמוּת הָאֱלׂהִים לׂא־חָשַׁב לוֹ לְשָׁלָל הֱיוֹתוֹ שָׁוֶה לֵאלׂהִים׃

ו

6

כִּי אִם־הִפְשִׁיט אֶת־עַצְמוֹ וַיִּלְבַּשׁ דְּמוּת עֶבֶד וַיְהִי דוֹמֶה לִבְנֵי אָדָם וַיִּמָּצֵא בִתְכוּנָתוֹ כְּבֶן אָדָם׃

ז

7

וַיַּשְׁפֵּל אֶת־נַפְשׁוֹ וַיִּכָּנַע עַד־מָוֶת עַד־מִיתַת הַצְּלִיבָה׃

ח

8

עַל־כֵּן הִגְבִּיהוֹ הָאֱלׂהִים מְאׂד וַיִּתֵּן־לוֹ שֵׁם נַעֲלֶה עַל־כָּל־שֵׁם׃

ט

9

אֲשֶׁר בְּשֵׁם יֵשׁוּעַ תִּכְרַע כָּל־בֶּרֶךְ אֲשֶׁר בַּשָּׁמַיִם וּבָאָרֶץ וּמִתַּחַת לָאָרֶץ׃

י

10

וְכָל־לָשׁוֹן תּוֹדֶה כִּי אָדוֹן יֵשׁוּעַ הַמָּשִׁיחַ לִכְבוֹד אֱלׂהִים הָאָב׃

יא

11

לָכֵן חֲבִיבַי כַּאֲשֶׁר שְׁמַעְתֶּם לִי בְּכָל־עֵת כֵּן לׂא בִּהְיוֹתִי אֶצְלְכֶם בִּלְבָד אֶלָּא־עוֹד יׂתֵר עַתָּה כְּשֶׁאֲנִי רָחוֹק מִכֶּם תִּיגְעוּ לִתְשׁוּעַת נַפְשׁׂתֵיכֶם בְּיִרְאָה וּבִרְעָדָה׃

יב

12

כִּי הָאֱלׂהִים הוּא הַפּׂעֵל בָּכֶם גַּם־לַחְפּׂץ גַּם־לִפְעׂל כְּפִי רְצוֹנוֹ׃

יג

13

עֲשׂוּ כָּל־דָּבָר בְּלׂא תְלֻנּוֹת וּבְלׂא מְזִמּוֹת׃

יד

14

לְמַעַן תִּהְיוּ נְקִיִּים וּטְהוֹרִים בְּנֵי אֱלׂהִים אֵין־מוּם בָּם בְּתוֹךְ דּוֹר עִקֵּשׁ וּפְתַלְתּׂל אֲשֶׁר־תָּאִירוּ בֵינֵיהֶם כִּמְאוֹרׂת בָּעוֹלָם׃

טו

15

מַחֲזִיקִים בִּדְבַר הַחַיִּים לִתְהִלָּה לִי בְּיוֹם הַמָּשִׁיחַ אֲשֶׁר לׂא לְחִנָּם רַצְתִּי וְלׂא לָרִיק יָגָעְתִּי׃

טז

16

אֲבָל גַּם כִּי־אֻסַּךְ עַל־זֶבַח אֱמוּנַתְכֶם וַעֲבוֹדָתָהּ הִנְנִי שָׂמֵחַ וְשָׂשׂ עִם־כֻּלְּכֶם׃

יז

17

וְכָכָה שִׂמְחוּ אַף־אַתֶּם וְשִׂישׂוּ עִמָּדִי׃

יח

18

וְקִוֵּיתִי בְּיֵשׁוּעַ אֲדׂנֵינוּ לִשְׁלׂחַ אֲלֵיכֶם בִּמְהֵרָה אֶת־טִימוֹתִיּוֹס לְמַעַן תָּנוּחַ דַּעְתִּי בְּהִוָּדַע לִי שְׁלוֹמְכֶם׃

יט

19

כִּי זוּלָתוֹ אֵין־אִתִּי אִישׁ כִּלְבָבִי אֲשֶׁר בְּלֵב שָׁלֵם יִדְאַג לָכֶם׃

כ

20

כִּי כֻלָּם אֵת אֲשֶׁר־לָהֶם יִדְרשׁוּ וְלׂא אֶת־אֲשֶׁר לְיֵשׁוּעַ הַמָּשִׁיחַ׃

כא

21

וְאוֹתוֹ יְדַעְתֶּם כִּי בָחוּן הוּא אֲשֶׁר כַּבֵּן הָעׂבֵד אֶת־אָבִיו כֵּן הָיָה אִתִּי בַּעֲבוֹדַת הַבְּשׂוֹרָה׃

כב

22

וְאׂתוֹ אֲקַוֶּה לִשְׁלׂחַ אֲלֵיכֶם מַהֵר כַּאֲשֶׁר אֶרְאֶה מַה־יִּהְיֶה לִי׃

כג

23

וּבָטַחְתִּי בַאֲדׂנֵינוּ אֲשֶׁר גַּם־אָנׂכִי אָבוֹא אֲלֵיכֶם בִּמְהֵרָה׃

כד

24

וְחָשַׁבְתִּי לְחוֹבָה לִי לִשְׁלׂחַ אֲלֵיכֶם אֶת־אֶפַּפְרוֹדִיטוֹס אָחִי וְעׂזְרִי וַחֲבֵרִי בַצָּבָא וְהוּא שְׁלִיחֲכֶם וּמְשָׁרֲתִי בִּצְרָכָי׃
